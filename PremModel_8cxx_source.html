126void PremModel::LoadModel(string filename)
127{
128
129 // Clear the current model
130 ClearModel();
131
132 // Use default if no file provided
133 if(filename == ""){
134 filename = PREM_DEFAULT;
135 }
136
137 // Open the file
138 ifstream fin;
139 fin.open(filename.c_str());
140
141 if(!fin){
142 cout << "ERROR: File " << filename << " not found!" << endl;
143 return;
144 }
145
146 // Variables for storing table rows
147 float radius, density, zoa, layer;
148
149 // Flag to mark we've passed the detector
150 bool crossed_det = false;
151
152 // Keep track of previous radius
153 double rprev = 0;
154
155 // Loop over table rows
156 while(fin >> radius >> density >> zoa >> layer){
157
158 // Radii must be ordered in model file
159 if(radius <= rprev){
160 cout << "ERROR: Radii are not sorted in increasing order in the model file" << endl;
161 ClearModel();
162 return;
163 }
164
165 // See if we passed the detector and decide whether
166 // to create a special layer for it
167 if(radius > fDetRadius - DET_TOL && !crossed_det){
168
169 crossed_det = true;
170
171 fDetLayer = fPremLayers.size();
172
173 // If detector is not near boundary, add a special layer
174 if(radius > fDetRadius + DET_TOL){
175 AddLayer(fDetRadius, density, zoa, layer);
176 } else if(radius != fDetRadius) {
177 //update detector radius
178 cout << "WARNING: Adjusting detector radius from " << fDetRadius << " km to " << radius << " km." << endl;
179 fDetRadius = radius;
180 }
181
182 }
183
184 // Add this layer to the model
185 AddLayer(radius, density, zoa, layer);
186
187 }
188
189 //Set the maximum radius in the model
190 fRadiusMax = fPremLayers.back().radius;
191
192}

324int PremModel::FillPath(double cosT, double phi)
325{
326
327 // Clear current path sequence
328 fNuPath.clear();
329
330 // Do nothing if cosine is unphysical
331 if(fabs(cosT) > 1) return 0;
332
333 // Define the minimum path radius
334 double minR = fDetRadius * sqrt(1 - cosT*cosT);
335 double minRsq = minR*minR;
336
337 // Set the top layer index
338 int toplayer = fPremLayers.size() - 1;
339
340 // Find the inner-most crossed layer
341 int minlayer = 0;
342 while(fPremLayers[minlayer].radius < minR) minlayer++;
343
344 // Compute the number of path segments needed
345 int nsteps = toplayer - fDetLayer;
346 if(cosT < 0) nsteps += 2*(fDetLayer - minlayer) + 1;
347
348 // Start at the top layer and go down
349 int layer = toplayer;
350 int dl = -1;
351
352 // Loop over all path segments
353 for(int i=0; i<nsteps; i++){
354
355 // Get square of the path length between this layer's
356 // outer radius and inner-most radius
357 double L1 = pow(fPremLayers[layer].radius,2) - minRsq;
358
359 // If L1 is negative, outer radius is not crossed.
360 // This only happens if detector is at the top layer and the
361 // neutrino is coming from above.
362 if(L1 < 0) return true;
363
364 // Get square of the path length between this layer's
365 // inner radius and inner-most radius
366 double L2 = -minRsq;
367 if(layer>0) L2 += pow(fPremLayers[layer-1].radius,2);
368
369 // If L2 is negative, inner radius is not crossed,
370 // so set this as the minimum layer.
371 bool ismin = (L2<=0 && cosT < 0);
372
373 // Store the path segment length
374 double dL;
375
376 // If it's the minimum layer, connect two outer radius
377 // crossing points. If not, compute difference between
378 // inner and outer radius crossings.
379 if(ismin) dL = 2 * sqrt(L1);
380 else if(L2>=0) dL = sqrt(L1) - sqrt(L2);
381 else dL = sqrt(L1); // This should never actually happen,
382 // but protect in case L2 is slightly
383 // negative when it should be zero.
384 // e.g. arriving at the detector from above.
385
386 // Add this path segment to the sequence
387 AddPath(dL, fPremLayers[layer]);
388
389 // If we reached the inner-most layer,
390 // start moving up again.
391 if(ismin) dl = 1;
392
393 // Move to next layer
394 layer += dl;
395
396 }
397
398 // Return the number of path segments
399 return fNuPath.size();
400
