80void PremModel::AddDetLayer(){
81
82 CleanIdentical();
83
84 fDetLayer = fPremLayers.size()-1;
85
86 if(fPremLayers.size() && fDetRadius > fPremLayers.back().radius){
87
88 cerr << "WARNING: Detector must be inside Earth model." << endl
89 << "WARNING: Adjusting detector radius from " << fDetRadius
90 << " km to " << fPremLayers.back().radius << " km." << endl;
91
92 fDetRadius = fPremLayers.back().radius;
93
94 return;
95
96 }
97
98 for(int i=0; i<fPremLayers.size(); i++){
99
100 double radius = fPremLayers[i].radius;
101
102 // See if we passed the detector and decide whether
103 // to create a special layer for it
104 if(radius > fDetRadius - DET_TOL){
105
106 fDetLayer = i;
107
108 // If detector is not near boundary, add a special layer
109 if(radius > fDetRadius + DET_TOL){
110 PremLayer det_layer = fPremLayers[i];
111 det_layer.radius = fDetRadius;
112 fPremLayers.insert(fPremLayers.begin()+fDetLayer, det_layer);
113 }
114 else if(radius != fDetRadius) {
115 //update detector radius
116 cerr << "WARNING: Adjusting detector radius from " << fDetRadius
117 << " km to " << radius << " km." << endl;
118 fDetRadius = radius;
119 }
120
121 break;
122
123 }
124
125 }

200void PremModel::LoadModel(string filename)
201{
202
203 // Clear the current model
204 ClearModel();
205
206 // Use default if no file provided
207 if(filename == ""){
208 filename = PREM_DEFAULT;
209 }
210
211 // Open the file
212 ifstream fin;
213 fin.open(filename.c_str());
214
215 if(!fin){
216 cerr << "ERROR: File " << filename << " not found!" << endl;
217 return;
218 }
219
220 // Variables for storing table rows
221 float radius, density, zoa, layer;
222
223 // Keep track of previous radius
224 double rprev = 0;
225
226 // Loop over table rows
227 while(fin >> radius >> density >> zoa >> layer){
228
229 // Radii must be ordered in model file
230 if(radius <= rprev){
231 cerr << "ERROR: Radii are not sorted in increasing order in the model file" << endl;
232 ClearModel();
233 return;
234 }
235
236 // Add this layer to the model
237 AddLayer(radius, density, zoa, layer);
238
239 }
240
241 AddDetLayer();
242
243 //Set the maximum radius in the model
244 fRadiusMax = fPremLayers.back().radius;
245
246}

378int PremModel::FillPath(double cosT, double phi)
379{
380
381 // Clear current path sequence
382 fNuPath.clear();
383
384 // Do nothing if cosine is unphysical
385 if(fabs(cosT) > 1) return 0;
386
387 // Define the minimum path radius
388 double minR = fDetRadius * sqrt(1 - cosT*cosT);
389 double minRsq = minR*minR;
390
391 // Set the top layer index
392 int toplayer = fPremLayers.size() - 1;
393
394 // Find the inner-most crossed layer
395 int minlayer = 0;
396 while(fPremLayers[minlayer].radius < minR) minlayer++;
397
398 // Compute the number of path segments needed
399 int nsteps = toplayer - fDetLayer;
400 if(cosT < 0) nsteps += 2*(fDetLayer - minlayer) + 1;
401
402 // Start at the top layer and go down
403 int layer = toplayer;
404 int dl = -1;
405
406 // Loop over all path segments
407 for(int i=0; i<nsteps; i++){
408
409 // Get square of the path length between this layer's
410 // outer radius and inner-most radius
411 double L1 = pow(fPremLayers[layer].radius,2) - minRsq;
412
413 // If L1 is negative, outer radius is not crossed.
414 // This only happens if detector is at the top layer and the
415 // neutrino is coming from above.
416 if(L1 < 0) return true;
417
418 // Get square of the path length between this layer's
419 // inner radius and inner-most radius
420 double L2 = -minRsq;
421 if(layer>0) L2 += pow(fPremLayers[layer-1].radius,2);
422
423 // If L2 is negative, inner radius is not crossed,
424 // so set this as the minimum layer.
425 bool ismin = (L2<=0 && cosT < 0);
426
427 // Store the path segment length
428 double dL;
429
430 // If it's the minimum layer, connect two outer radius
431 // crossing points. If not, compute difference between
432 // inner and outer radius crossings.
433 if(ismin) dL = 2 * sqrt(L1);
434 else if(L2>=0) dL = sqrt(L1) - sqrt(L2);
435 else dL = sqrt(L1); // This should never actually happen,
436 // but protect in case L2 is slightly
437 // negative when it should be zero.
438 // e.g. arriving at the detector from above.
439
440 // Add this path segment to the sequence
441 AddPath(dL, fPremLayers[layer]);
442
443 // If we reached the inner-most layer,
444 // start moving up again.
445 if(ismin) dl = 1;
446
447 // Move to next layer
448 layer += dl;
449
450 }
451
452 // Return the number of path segments
453 return fNuPath.size();
454
