32void PMNS_Iter::SetExpVL(NuPath p)
33{
34
35 double kr2GNe = kK2*M_SQRT2*kGf;
36 kr2GNe *= p.density * p.zoa; // Matter potential in eV
37
38 fVL = kr2GNe * kKm2eV * p.length;
39
40 fExpVL = complexD(cos(fVL), -sin(fVL));
41 if(fIsNuBar) fExpVL = conj(fExpVL);

102void PMNS_Iter::SplitPropagate(NuPath p)
103{
104
105 SetExpVL(p);
106
107 double dm = 0;
108 for(int i=0; i<fNumNus; i++){
109 if(dm<fabs(fDm[i])) dm = fabs(fDm[i]);
110 }
111 dm *= kKm2eV * p.length / (2 * kGeV2eV * fEnergy);
112
113 int nsplit = sqrt(dm * fVL / fPrec);
114 nsplit++;
115
116 p.length /= nsplit;
117 SetExpVL(p);
118
119 for(int i=0; i<nsplit; i++){
120
121 PropagatePath(p);
122 PropMatter();
123
124 }
125
126}

134void PMNS_Iter::SetVacuumEigensystem()
135{
136
137 if(!fBuiltHms){
138
139 double s12, s23, s13, c12, c23, c13;
140 complexD idelta(0.0, fDelta[0][2]);
141 if(fIsNuBar) idelta = conj(idelta);
142
143 s12 = sin(fTheta[0][1]); s23 = sin(fTheta[1][2]); s13 = sin(fTheta[0][2]);
144 c12 = cos(fTheta[0][1]); c23 = cos(fTheta[1][2]); c13 = cos(fTheta[0][2]);
145
146 fEvec[0][0] = c12*c13;
147 fEvec[0][1] = s12*c13;
148 fEvec[0][2] = s13*exp(-idelta);
149
150 fEvec[1][0] = -s12*c23-c12*s23*s13*exp(idelta);
151 fEvec[1][1] = c12*c23-s12*s23*s13*exp(idelta);
152 fEvec[1][2] = s23*c13;
153
154 fEvec[2][0] = s12*s23-c12*c23*s13*exp(idelta);
155 fEvec[2][1] = -c12*s23-s12*c23*s13*exp(idelta);
156 fEvec[2][2] = c23*c13;
157
158 }
159
160 fBuiltHms = true;
161
162 fEval[0] = 0;
163 fEval[1] = fDm[1] / (2 * kGeV2eV*fEnergy);
164 fEval[2] = fDm[2] / (2 * kGeV2eV*fEnergy);
165
166}
