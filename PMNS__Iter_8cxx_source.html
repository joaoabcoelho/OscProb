47void PMNS_Iter::SetExpVL(NuPath p)
48{
49 double kr2GNe = kK2 * M_SQRT2 * kGf;
50 kr2GNe *= p.density * p.zoa; // Matter potential in eV
51
52 fVL = kr2GNe * kKm2eV * p.length;
53
54 fExpVL = complexD(cos(fVL), -sin(fVL));
55 if (fIsNuBar) fExpVL = conj(fExpVL);
56}

68void PMNS_Iter::SolveHam()
69{
70 // Do vacuum oscillation
71 if (!fBuiltHms) {
72 PMNS_Fast::SetVacuumEigensystem();
73 fBuiltHms = true;
74 fGotES = true;
75 fPrevEnergy = fEnergy;
76 return;
77 }
78
79 if (fGotES) return;
80
81 for (int i = 1; i < fNumNus; i++) { fEval[i] *= fPrevEnergy / fEnergy; }
82
83 fPrevEnergy = fEnergy;
84
85 fGotES = true;
86 return;
87}

103void PMNS_Iter::PropagatePath(NuPath p)
104{
105 SetExpVL(p);
106
107 double dm = 0;
108 for (int i = 0; i < fNumNus; i++) {
109 if (dm < fabs(fDm[i])) dm = fabs(fDm[i]);
110 }
111 dm *= kKm2eV * p.length / (2 * kGeV2eV * fEnergy);
112
113 int nsplit = sqrt(0.065 * dm * fVL / fPrec);
114 if (nsplit > 1) {
115 p.length /= nsplit;
116 SetExpVL(p);
117 }
118 else
119 nsplit = 1;
120
121 for (int i = 0; i < nsplit; i++) {
122 PMNS_Base::PropagatePath(p);
123 PropMatter();
124 }
125}
