118 SetTestPath(p);
119

121
122 int ntests = 0;

124
125 TCanvas* c1 = 0;

127 TH1D* h = 0;
128
129 for(int flvi=0; flvi<3; flvi++){
130 for(int flvf=0; flvf<3; flvf++){
131 for(int isnb=0; isnb<2; isnb++){
132 p->SetIsNuBar(isnb);
133 TString hname = TString::Format("h%d%d%d",flvi,flvf,isnb);
134 h0 = (TH1D*)f->Get(hname);
135 h = (TH1D*)h0->Clone();
136 bool plot = false;
137 for(int i=1; i<=h0->GetNbinsX(); i++){
138 double energy = h->GetBinCenter(i);
139 double dE = h->GetBinWidth(i);
140 double p0 = h0->GetBinContent(i);
141 double p1 = p->AvgProb(flvi, flvf, energy, dE);
142 ntests++;
143 if(abs(p0-p1)>1e-12){
144 plot = true;
145 fails++;
146 }
147 h->SetBinContent(i, p1);
148 }
149 if(plot){
150 c1 = new TCanvas();
151 c1->Divide(1,2);
152 c1->cd(1);
153 SetHist(h0, kBlue);
154 SetHist(h, kRed);
155 TString nu_lab = isnb ? "#bar{#nu}" : "#nu";
156 TString flv_lab[3] = {"e","#mu","#tau"};
157 TString ylab = "P(" + nu_lab + "_{" + flv_lab[flvi] +
158 "}#rightarrow" + nu_lab + "_{" + flv_lab[flvf] + "})";
159 h0->SetTitle(";Energy [GeV];"+ylab+";");
160 h->SetLineStyle(7);
161 double ymax = max(h->GetMaximum(),
162 h0->GetMaximum());
163 double ymin = min(h->GetMinimum(),
164 h0->GetMinimum());
165 h0->GetYaxis()->SetRangeUser(ymin, ymax);
166 h0->DrawCopy("hist");
167 h->DrawCopy("hist same");
168 SetTH1Margin();
169 gPad->SetLogx();
170 c1->cd(2);
171 h->SetTitle(";Energy [GeV];#Delta"+ylab+";");
172 h->Add(h0,-1);
173 h->SetLineStyle(1);
174 h->DrawCopy("hist");
175 SetTH1Margin();
176 gPad->SetLogx();
177 MiscText(0.55,0.8,0.1,filename,kGray,1);
178 c1->DrawClone();
179 TString pngfile = filename;
180 pngfile.ReplaceAll(".root",".png");
181 c1->SaveAs("plots/Failed_"+hname+"_"+pngfile);
182 delete c1;
183 }
184 delete h0, h;
185 }}}
186
187 if(fails>0){
188 printf("FAILED: Found %d differences in %d tests (%.3g%%) in %s\n",
189 fails, ntests, 100.*fails/ntests, filename.Data());
190 }
191 else cout << "PASSED: No differences found in " << filename << endl;
192
193 return fails;
