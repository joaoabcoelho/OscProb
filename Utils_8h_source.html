157int CheckProb(OscProb::PMNS_Base* p, TString filename){
158
159 SetTestPath(p);
160
161 TFile* f = new TFile("data/"+filename, "read");
162
163 int ntests = 0;
164 int fails = 0;
165
166 TCanvas* c1 = 0;
167 TH1D* h0 = 0;
168 TH1D* h = 0;
169
170 for(int flvi=0; flvi<3; flvi++){
171 for(int flvf=0; flvf<3; flvf++){
172 for(int isnb=0; isnb<2; isnb++){
173 p->SetIsNuBar(isnb);
174 TString hname = TString::Format("h%d%d%d",flvi,flvf,isnb);
175 h0 = (TH1D*)f->Get(hname);
176 h = (TH1D*)h0->Clone();
177 bool plot = false;
178 for(int i=1; i<=h0->GetNbinsX(); i++){
179 double energy = h->GetBinCenter(i);
180 double dE = h->GetBinWidth(i);
181 double p0 = h0->GetBinContent(i);
182 double p1 = p->AvgProb(flvi, flvf, energy, dE);
183 ntests++;
184 if(abs(p0-p1)>1e-12){
185 plot = true;
186 fails++;
187 }
188 h->SetBinContent(i, p1);
189 }
190 if(plot){
191 c1 = new TCanvas();
192 c1->Divide(1,2);
193 c1->cd(1);
194 SetHist(h0, kBlue);
195 SetHist(h, kRed);
196 TString nu_lab = isnb ? "#bar{#nu}" : "#nu";
197 TString flv_lab[3] = {"e","#mu","#tau"};
198 TString ylab = "P(" + nu_lab + "_{" + flv_lab[flvi] +
199 "}#rightarrow" + nu_lab + "_{" + flv_lab[flvf] + "})";
200 h0->SetTitle(";Energy [GeV];"+ylab+";");
201 h->SetLineStyle(7);
202 double ymax = max(h->GetMaximum(),
203 h0->GetMaximum());
204 double ymin = min(h->GetMinimum(),
205 h0->GetMinimum());
206 h0->GetYaxis()->SetRangeUser(ymin, ymax);
207 h0->DrawCopy("hist");
208 h->DrawCopy("hist same");
209 SetTH1Margin();
210 gPad->SetLogx();
211 c1->cd(2);
212 h->SetTitle(";Energy [GeV];#Delta"+ylab+";");
213 h->Add(h0,-1);
214 h->SetLineStyle(1);
215 h->DrawCopy("hist");
216 SetTH1Margin();
217 gPad->SetLogx();
218 MiscText(0.55,0.8,0.1,filename,kGray,1);
219 c1->DrawClone();
220 TString pngfile = filename;
221 pngfile.ReplaceAll(".root",".png");
222 c1->SaveAs("plots/Failed_"+hname+"_"+pngfile);
223 delete c1;
224 }
225 delete h0, h;
226 }}}
227
228 if(fails>0){
229 printf("FAILED: Found %d differences in %d tests (%.3g%%) in %s\n",
230 fails, ntests, 100.*fails/ntests, filename.Data());
231 }
232 else cout << "PASSED: No differences found in " << filename << endl;
233
234 return fails;
