137bool PMNS_Base::TryCache()
138{
139
140 if(fUseCache && !fMixCache.empty()){
141
142 fProbe.SetVars(fEnergy, fPath, fIsNuBar);
143
144 std::set<EigenPoint>::iterator it = fMixCache.find(fProbe);
145
146 if(it != fMixCache.end()){
147 for(int i=0; i<fNumNus; i++){
148 fEval[i] = (*it).fEval[i] * (*it).fEnergy / fEnergy;
149 for(int j=0; j<fNumNus; j++){
150 fEvec[i][j] = (*it).fEvec[i][j];
151 }
152 }
153 return true;
154 }
155
156 }
157
158 return false;
159
160}

492void PMNS_Base::SetAtt(std::vector<double> att, int idx){
493
494 // Get the sizes of the attribute and
495 // path sequence vectors
496 int nA = att.size();
497 int nP = fNuPaths.size();
498
499 // If the vector sizes are equal, update this attribute
500 if(nA == nP){
501
502 for(int i=0; i<nP; i++){
503
504 switch(idx){
505 case 0: fNuPaths[i].length = att[i]; break;
506 case 1: fNuPaths[i].density = att[i]; break;
507 case 2: fNuPaths[i].zoa = att[i]; break;
508 case 3: fNuPaths[i].layer = att[i]; break;
509 }
510
511 }
512
513 }
514 // If the vector sizes differ, create a new path sequence
515 // and set value for this attribute. Other attributes will
516 // be taken from default single path.
517 else{
518
519 cout << "Warning: New vector size. Starting new path vector." << endl;
520 cout << "To avoid possible issues, use the SetPath function." << endl;
521
522 // Start a new standard path just
523 // to set default values
524 SetStdPath();
525
526 // Create a path segment with default values
527 NuPath p = fNuPaths[0];
528
529 // Clear the path sequence
530 ClearPath();
531
532 // Set this particular attribute's value
533 // and add the path segment to the sequence
534 for(int i=0; i<nA; i++){
535
536 switch(idx){
537 case 0: p.length = att[i]; break;
538 case 1: p.density = att[i]; break;
539 case 2: p.zoa = att[i]; break;
540 case 3: p.layer = att[i]; break;
541 }
542
543 AddPath(p);
544
545 }

851//......................................................................
857void PMNS_Base::RotateState(int i, int j){
858
859 // Do nothing if angle is zero
860 if(fTheta[i][j]==0) return;
861
862 double sij = sin(fTheta[i][j]);
863 double cij = cos(fTheta[i][j]);
864
865 complexD buffer;
866
867 if(i+1==j || fDelta[i][j]==0){
868 buffer = cij*fNuState[i] + sij*fNuState[j];
869 fNuState[j] = cij*fNuState[j] - sij*fNuState[i];
870 }
871 else {
872 complexD eij = complexD(cos(fDelta[i][j]), -sin(fDelta[i][j]));
873 buffer = cij*fNuState[i] + sij*eij*fNuState[j];
874 fNuState[j] = cij*fNuState[j] - sij*conj(eij)*fNuState[i];
875 }
876
877 fNuState[i] = buffer;
878
879}

922void PMNS_Base::RotateH(int i,int j,std::vector< std::vector<complexD> >& Ham){
923
924 // Do nothing if angle is zero
925 if(fTheta[i][j]==0) return;
926
927 double fSinBuffer = sin(fTheta[i][j]);
928 double fCosBuffer = cos(fTheta[i][j]);
929
930 double fHmsBufferD;
931 complexD fHmsBufferC;
932
933 // With Delta
934 if(i+1<j){
935 complexD fExpBuffer = complexD(cos(fDelta[i][j]), -sin(fDelta[i][j]));
936
937 // General case
938 if(i>0){
939 // Top columns
940 for(int k=0; k<i; k++){
941 fHmsBufferC = Ham[k][i];
942
943 Ham[k][i] *= fCosBuffer;
944 Ham[k][i] += Ham[k][j] * fSinBuffer * conj(fExpBuffer);
945
946 Ham[k][j] *= fCosBuffer;
947 Ham[k][j] -= fHmsBufferC * fSinBuffer * fExpBuffer;
948 }
949
950 // Middle row and column
951 for(int k=i+1; k<j; k++){
952 fHmsBufferC = Ham[k][j];
953
954 Ham[k][j] *= fCosBuffer;
955 Ham[k][j] -= conj(Ham[i][k]) * fSinBuffer * fExpBuffer;
956
957 Ham[i][k] *= fCosBuffer;
958 Ham[i][k] += fSinBuffer * fExpBuffer * conj(fHmsBufferC);
959 }
960
961 // Nodes ij
962 fHmsBufferC = Ham[i][i];
963 fHmsBufferD = real(Ham[j][j]);
964
965 Ham[i][i] *= fCosBuffer * fCosBuffer;
966 Ham[i][i] += 2 * fSinBuffer * fCosBuffer * real(Ham[i][j] * conj(fExpBuffer));
967 Ham[i][i] += fSinBuffer * Ham[j][j] * fSinBuffer;
968
969 Ham[j][j] *= fCosBuffer * fCosBuffer;
970 Ham[j][j] += fSinBuffer * fHmsBufferC * fSinBuffer;
971 Ham[j][j] -= 2 * fSinBuffer * fCosBuffer * real(Ham[i][j] * conj(fExpBuffer));
972
973 Ham[i][j] -= 2 * fSinBuffer * real(Ham[i][j] * conj(fExpBuffer)) * fSinBuffer * fExpBuffer;
974 Ham[i][j] += fSinBuffer * fCosBuffer * (fHmsBufferD - fHmsBufferC) * fExpBuffer;
975
976 }
977 // First rotation on j (No top columns)
978 else{
979 // Middle rows and columns
980 for(int k=i+1; k<j; k++){
981 Ham[k][j] = -conj(Ham[i][k]) * fSinBuffer * fExpBuffer;
982
983 Ham[i][k] *= fCosBuffer;
984 }
985
986 // Nodes ij
987 fHmsBufferD = real(Ham[i][i]);
988
989 Ham[i][j] = fSinBuffer * fCosBuffer * (Ham[j][j] - fHmsBufferD) * fExpBuffer;
990
991 Ham[i][i] *= fCosBuffer * fCosBuffer;
992 Ham[i][i] += fSinBuffer * Ham[j][j] * fSinBuffer;
993
994 Ham[j][j] *= fCosBuffer * fCosBuffer;
995 Ham[j][j] += fSinBuffer * fHmsBufferD * fSinBuffer;
996 }
997
998 }
999 // Without Delta (No middle rows or columns: j = i+1)
1000 else{
1001 // General case
1002 if(i>0){
1003 // Top columns
1004 for(int k=0; k<i; k++){
1005 fHmsBufferC = Ham[k][i];
1006
1007 Ham[k][i] *= fCosBuffer;
1008 Ham[k][i] += Ham[k][j] * fSinBuffer;
1009
1010 Ham[k][j] *= fCosBuffer;
1011 Ham[k][j] -= fHmsBufferC * fSinBuffer;
1012 }
1013
1014 // Nodes ij
1015 fHmsBufferC = Ham[i][i];
1016 fHmsBufferD = real(Ham[j][j]);
1017
1018 Ham[i][i] *= fCosBuffer * fCosBuffer;
1019 Ham[i][i] += 2 * fSinBuffer * fCosBuffer * real(Ham[i][j]);
1020 Ham[i][i] += fSinBuffer * Ham[j][j] * fSinBuffer;
1021
1022 Ham[j][j] *= fCosBuffer * fCosBuffer;
1023 Ham[j][j] += fSinBuffer * fHmsBufferC * fSinBuffer;
1024 Ham[j][j] -= 2 * fSinBuffer * fCosBuffer * real(Ham[i][j]);
1025
1026 Ham[i][j] -= 2 * fSinBuffer * real(Ham[i][j]) * fSinBuffer;
1027 Ham[i][j] += fSinBuffer * fCosBuffer * (fHmsBufferD - fHmsBufferC);
1028
1029 }
1030 // First rotation (theta12)
1031 else{
1032
1033 Ham[i][j] = fSinBuffer * fCosBuffer * Ham[j][j];
1034
1035 Ham[i][i] = fSinBuffer * Ham[j][j] * fSinBuffer;
1036
1037 Ham[j][j] *= fCosBuffer * fCosBuffer;
1038
1039 }
1040 }
1041
1042}

1056void PMNS_Base::BuildHms()
1057{
1058
1059 // Check if anything changed
1060 if(fBuiltHms) return;
1061
1062 // Tag to recompute eigensystem
1063 fGotES = false;
1064
1065 for(int j=0; j<fNumNus; j++){
1066 // Set mass splitting
1067 fHms[j][j] = fDm[j];
1068 // Reset off-diagonal elements
1069 for(int i=0; i<j; i++){
1070 fHms[i][j] = 0;
1071 }
1072 // Rotate j neutrinos
1073 for(int i=0; i<j; i++){
1074 RotateH(i,j,fHms);
1075 }
1076 }
1077
1078 ClearCache();
1079
1080 // Tag as built
1081 fBuiltHms = true;
1082
1083}

1090void PMNS_Base::PropagatePath(NuPath p)
1091{
1092
1093 // Set the neutrino path
1094 SetCurPath(p);
1095
1096 // Solve for eigensystem
1097 SolveHam();
1098
1099 double LengthIneV = kKm2eV * p.length;
1100 for(int i=0; i<fNumNus; i++){
1101 double arg = fEval[i] * LengthIneV;
1102 fPhases[i] = complexD(cos(arg), -sin(arg));
1103 }
1104
1105 for(int i=0;i<fNumNus;i++){
1106 fBuffer[i] = 0;
1107 for(int j=0;j<fNumNus;j++){
1108 fBuffer[i] += conj(fEvec[j][i]) * fNuState[j];
1109 }
1110 fBuffer[i] *= fPhases[i];
1111 }
1112
1113 // Propagate neutrino state
1114 for(int i=0;i<fNumNus;i++){
1115 fNuState[i] = 0;
1116 for(int j=0;j<fNumNus;j++){

1397double PMNS_Base::AvgProb(std::vector<complexD> nu_in, int flvf, double E, double dE)
1398{
1399
1400 // Do nothing if energy is not positive
1401 if(E<=0) return 0;
1402
1403 if(fNuPaths.empty()) return 0;
1404
1405 // Don't average zero width
1406 if(dE<=0) return Prob(nu_in, flvf, E);
1407
1408 // Make sure fPath is set
1409 // Use average if multiple paths
1410 SetCurPath(AvgPath(fNuPaths));
1411
1412 // Define L/E variables
1413 double LoE = 0;
1414 double dLoE = 0;
1415
1416 // Set a minimum energy
1417 double minE = 0.1 * E;
1418
1419 // Transform range to L/E
1420 // Full range if low edge > minE
1421 if(E-dE/2 > minE){
1422 LoE = 0.5 * (fPath.length/(E-dE/2) + fPath.length/(E+dE/2));
1423 dLoE = fPath.length/(E-dE/2) - fPath.length/(E+dE/2);
1424 }
1425 // Else start at minE
1426 else{
1427 LoE = 0.5 * (fPath.length/minE + fPath.length/(E+dE/2));
1428 dLoE = fPath.length/minE - fPath.length/(E+dE/2);
1429 }
1430
1431 // Compute average in LoE
1432 return AvgProbLoE(nu_in, flvf, LoE, dLoE);

1497double PMNS_Base::AvgProbLoE(std::vector<complexD> nu_in, int flvf, double LoE, double dLoE)
1498{
1499
1500 // Do nothing if L/E is not positive
1501 if(LoE<=0) return 0;
1502
1503 if(fNuPaths.empty()) return 0;
1504
1505 // Make sure fPath is set
1506 // Use average if multiple paths
1507 SetCurPath(AvgPath(fNuPaths));
1508
1509 // Set the energy at bin center
1510 SetEnergy(fPath.length/LoE);
1511
1512 // Don't average zero width
1513 if(dLoE<=0) return Prob(nu_in, flvf);
1514
1515 // Get sample points for this bin
1516 vector<double> samples = GetSamplePoints(LoE, dLoE);
1517
1518 // Variables to fill sample
1519 // probabilities and weights
1520 double sumw = 0;
1521 double prob = 0;
1522 double length = fPath.length;
1523
1524 // Loop over all sample points
1525 for(int j=0; j<int(samples.size()); j++){
1526
1527 // Set (L/E)^-2 weights
1528 double w = 1./pow(samples[j],2);
1529
1530 // Add weighted probability
1531 prob += w * Prob(nu_in, flvf, length / samples[j]);
1532
1533 // Increment sum of weights
1534 sumw += w;
1535
1536 }
1537
1538 // Return weighted average of probabilities
1539 return prob / sumw;

1556vector<double> PMNS_Base::GetSamplePoints(double LoE, double dLoE)
1557{
1558
1559 // Solve Hamiltonian to get eigenvalues
1560 SolveHam();
1561
1562 // Define conversion factor [km/GeV -> 1/(4 eV^2)]
1563 const double k1267 = kKm2eV / (4 * kGeV2eV);
1564
1565 // Get list of all effective Dm^2
1566 vector<double> effDm;
1567
1568 for(int i=0; i<fNumNus-1; i++){
1569 for(int j=i+1; j<fNumNus; j++){
1570 effDm.push_back( 2 * kGeV2eV * fEnergy * fabs(fEval[j] - fEval[i]) );
1571 }
1572 }
1573
1574 int numDm = effDm.size();
1575
1576 // Sort the effective Dm^2 list
1577 sort(effDm.begin(), effDm.end());
1578
1579 // Set a number of sub-divisions to achieve "good" accuracy
1580 // This needs to be studied better
1581 int n_div = ceil( 20 * pow(dLoE/LoE,0.8) );
1582 //int n_div = 1;
1583
1584 // A vector to store sample points
1585 vector<double> allSamples;
1586
1587 // Loop over sub-divisions
1588 for(int k=0; k<n_div; k++){
1589
1590 // Define sub-division center and width
1591 double bctr = LoE - dLoE/2 + (k+0.5)*dLoE/n_div;
1592 double bwdt = dLoE/n_div;
1593
1594 // Make a vector of L/E sample values
1595 // Initialized in the sub-division center
1596 vector<double> samples;
1597 samples.push_back(bctr);
1598
1599 // Loop over all Dm^2 to average each frequency
1600 // This will recursively sample points in smaller
1601 // bins so that all relevant frequencies are used
1602 for(int i=0; i<numDm; i++){
1603
1604 // Copy the list of sample L/E values
1605 vector<double> prev = samples;
1606
1607 // Redefine bin width to lie within full sub-division
1608 double Width = 2*min(prev[0] - (bctr - bwdt/2), (bctr + bwdt/2) - prev[0]);
1609
1610 // Compute oscillation argument sorted from lowest to highest
1611 const double arg = k1267 * effDm[i] * Width;
1612
1613 // Skip small oscillation values.
1614 // If it's the last one, lower the tolerance
1615 if(i < numDm-1){
1616 if(arg<0.9) continue;
1617 }
1618 else{
1619 if(arg<0.1) continue;
1620 }
1621
1622 // Reset samples to redefine them
1623 samples.clear();
1624
1625 // Loop over previous samples
1626 for(int j=0; j<int(prev.size()); j++){
1627
1628 // Compute new sample points around old samples
1629 // This is based on a oscillatory quadrature rule
1630 double sample = (1/sqrt(3)) * (Width/2);
1631 if(arg!=0) sample = acos(sin(arg)/arg)/arg * (Width/2);
1632
1633 // Add samples above and below center
1634 samples.push_back(prev[j]-sample);
1635 samples.push_back(prev[j]+sample);
1636
1637 }
1638
1639 }// End of loop over Dm^2
1640
1641 // Add sub-division samples to the end of allSamples vector
1642 allSamples.insert(allSamples.end(), samples.begin(), samples.end());
1643
1644 }// End of loop over sub-divisions
1645
1646 // Return all sample points
1647 return allSamples;
1648
1649}
