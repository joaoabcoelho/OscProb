144bool PMNS_Base::TryCache()
145{
146
147 if(fUseCache && !fMixCache.empty()){
148
149 fProbe.SetVars(fEnergy, fPath, fIsNuBar);
150
151 unordered_set<EigenPoint>::iterator it = fMixCache.find(fProbe);
152
153 if(it != fMixCache.end()){
154 for(int i=0; i<fNumNus; i++){
155 fEval[i] = (*it).fEval[i] * (*it).fEnergy / fEnergy;
156 for(int j=0; j<fNumNus; j++){
157 fEvec[i][j] = (*it).fEvec[i][j];
158 }
159 }
160 return true;
161 }
162
163 }
164
165 return false;
166

499void PMNS_Base::SetAtt(vectorD att, int idx){
500
501 // Get the sizes of the attribute and
502 // path sequence vectors
503 int nA = att.size();
504 int nP = fNuPaths.size();
505
506 // If the vector sizes are equal, update this attribute
507 if(nA == nP){
508
509 for(int i=0; i<nP; i++){
510
511 switch(idx){
512 case 0: fNuPaths[i].length = att[i]; break;
513 case 1: fNuPaths[i].density = att[i]; break;
514 case 2: fNuPaths[i].zoa = att[i]; break;
515 case 3: fNuPaths[i].layer = att[i]; break;
516 }
517
518 }
519
520 }
521 // If the vector sizes differ, create a new path sequence
522 // and set value for this attribute. Other attributes will
523 // be taken from default single path.
524 else{
525
526 cerr << "WARNING: New vector size. Starting new path vector." << endl
527 << "To avoid possible issues, use the SetPath function." << endl;
528
529 // Start a new standard path just
530 // to set default values
531 SetStdPath();
532
533 // Create a path segment with default values
534 NuPath p = fNuPaths[0];
535
536 // Clear the path sequence
537 ClearPath();
538
539 // Set this particular attribute's value
540 // and add the path segment to the sequence
541 for(int i=0; i<nA; i++){
542
543 switch(idx){
544 case 0: p.length = att[i]; break;
545 case 1: p.density = att[i]; break;
546 case 2: p.zoa = att[i]; break;
547 case 3: p.layer = att[i]; break;

834double PMNS_Base::GetDmEff(int j)
835{

864void PMNS_Base::RotateState(int i, int j){
865
866 // Do nothing if angle is zero
867 if(fTheta[i][j]==0) return;
868
869 double sij = sin(fTheta[i][j]);
870 double cij = cos(fTheta[i][j]);
871
872 complexD buffer;
873
874 if(i+1==j || fDelta[i][j]==0){
875 buffer = cij*fNuState[i] + sij*fNuState[j];
876 fNuState[j] = cij*fNuState[j] - sij*fNuState[i];
877 }
878 else {
879 complexD eij = complexD(cos(fDelta[i][j]), -sin(fDelta[i][j]));
880 buffer = cij*fNuState[i] + sij*eij*fNuState[j];
881 fNuState[j] = cij*fNuState[j] - sij*conj(eij)*fNuState[i];
882 }
883
884 fNuState[i] = buffer;
885
886}

929void PMNS_Base::RotateH(int i,int j, matrixC& Ham){
930
931 // Do nothing if angle is zero
932 if(fTheta[i][j]==0) return;
933
934 double fSinBuffer = sin(fTheta[i][j]);
935 double fCosBuffer = cos(fTheta[i][j]);
936
937 double fHmsBufferD;
938 complexD fHmsBufferC;
939
940 // With Delta
941 if(i+1<j){
942 complexD fExpBuffer = complexD(cos(fDelta[i][j]), -sin(fDelta[i][j]));
943
944 // General case
945 if(i>0){
946 // Top columns
947 for(int k=0; k<i; k++){
948 fHmsBufferC = Ham[k][i];
949
950 Ham[k][i] *= fCosBuffer;
951 Ham[k][i] += Ham[k][j] * fSinBuffer * conj(fExpBuffer);
952
953 Ham[k][j] *= fCosBuffer;
954 Ham[k][j] -= fHmsBufferC * fSinBuffer * fExpBuffer;
955 }
956
957 // Middle row and column
958 for(int k=i+1; k<j; k++){
959 fHmsBufferC = Ham[k][j];
960
961 Ham[k][j] *= fCosBuffer;
962 Ham[k][j] -= conj(Ham[i][k]) * fSinBuffer * fExpBuffer;
963
964 Ham[i][k] *= fCosBuffer;
965 Ham[i][k] += fSinBuffer * fExpBuffer * conj(fHmsBufferC);
966 }
967
968 // Nodes ij
969 fHmsBufferC = Ham[i][i];
970 fHmsBufferD = real(Ham[j][j]);
971
972 Ham[i][i] *= fCosBuffer * fCosBuffer;
973 Ham[i][i] += 2 * fSinBuffer * fCosBuffer * real(Ham[i][j] * conj(fExpBuffer));
974 Ham[i][i] += fSinBuffer * Ham[j][j] * fSinBuffer;
975
976 Ham[j][j] *= fCosBuffer * fCosBuffer;
977 Ham[j][j] += fSinBuffer * fHmsBufferC * fSinBuffer;
978 Ham[j][j] -= 2 * fSinBuffer * fCosBuffer * real(Ham[i][j] * conj(fExpBuffer));
979
980 Ham[i][j] -= 2 * fSinBuffer * real(Ham[i][j] * conj(fExpBuffer)) * fSinBuffer * fExpBuffer;
981 Ham[i][j] += fSinBuffer * fCosBuffer * (fHmsBufferD - fHmsBufferC) * fExpBuffer;
982
983 }
984 // First rotation on j (No top columns)
985 else{
986 // Middle rows and columns
987 for(int k=i+1; k<j; k++){
988 Ham[k][j] = -conj(Ham[i][k]) * fSinBuffer * fExpBuffer;
989
990 Ham[i][k] *= fCosBuffer;
991 }
992
993 // Nodes ij
994 fHmsBufferD = real(Ham[i][i]);
995
996 Ham[i][j] = fSinBuffer * fCosBuffer * (Ham[j][j] - fHmsBufferD) * fExpBuffer;
997
998 Ham[i][i] *= fCosBuffer * fCosBuffer;
999 Ham[i][i] += fSinBuffer * Ham[j][j] * fSinBuffer;
1000
1001 Ham[j][j] *= fCosBuffer * fCosBuffer;
1002 Ham[j][j] += fSinBuffer * fHmsBufferD * fSinBuffer;
1003 }
1004
1005 }
1006 // Without Delta (No middle rows or columns: j = i+1)
1007 else{
1008 // General case
1009 if(i>0){
1010 // Top columns
1011 for(int k=0; k<i; k++){
1012 fHmsBufferC = Ham[k][i];
1013
1014 Ham[k][i] *= fCosBuffer;
1015 Ham[k][i] += Ham[k][j] * fSinBuffer;
1016
1017 Ham[k][j] *= fCosBuffer;
1018 Ham[k][j] -= fHmsBufferC * fSinBuffer;
1019 }
1020
1021 // Nodes ij
1022 fHmsBufferC = Ham[i][i];
1023 fHmsBufferD = real(Ham[j][j]);
1024
1025 Ham[i][i] *= fCosBuffer * fCosBuffer;
1026 Ham[i][i] += 2 * fSinBuffer * fCosBuffer * real(Ham[i][j]);
1027 Ham[i][i] += fSinBuffer * Ham[j][j] * fSinBuffer;
1028
1029 Ham[j][j] *= fCosBuffer * fCosBuffer;
1030 Ham[j][j] += fSinBuffer * fHmsBufferC * fSinBuffer;
1031 Ham[j][j] -= 2 * fSinBuffer * fCosBuffer * real(Ham[i][j]);
1032
1033 Ham[i][j] -= 2 * fSinBuffer * real(Ham[i][j]) * fSinBuffer;
1034 Ham[i][j] += fSinBuffer * fCosBuffer * (fHmsBufferD - fHmsBufferC);
1035
1036 }
1037 // First rotation (theta12)
1038 else{
1039
1040 Ham[i][j] = fSinBuffer * fCosBuffer * Ham[j][j];
1041
1042 Ham[i][i] = fSinBuffer * Ham[j][j] * fSinBuffer;
1043
1044 Ham[j][j] *= fCosBuffer * fCosBuffer;
1045
1046 }
1047 }
1048
1049}

1063void PMNS_Base::BuildHms()
1064{
1065
1066 // Check if anything changed
1067 if(fBuiltHms) return;
1068
1069 // Tag to recompute eigensystem
1070 fGotES = false;
1071
1072 for(int j=0; j<fNumNus; j++){
1073 // Set mass splitting
1074 fHms[j][j] = fDm[j];
1075 // Reset off-diagonal elements
1076 for(int i=0; i<j; i++){
1077 fHms[i][j] = 0;
1078 }
1079 // Rotate j neutrinos
1080 for(int i=0; i<j; i++){
1081 RotateH(i,j,fHms);
1082 }
1083 }
1084
1085 ClearCache();
1086
1087 // Tag as built
1088 fBuiltHms = true;
1089

1097void PMNS_Base::PropagatePath(NuPath p)
1098{
1099
1100 // Set the neutrino path
1101 SetCurPath(p);
1102
1103 // Solve for eigensystem
1104 SolveHam();
1105
1106 double LengthIneV = kKm2eV * p.length;
1107 for(int i=0; i<fNumNus; i++){
1108 double arg = fEval[i] * LengthIneV;
1109 fPhases[i] = complexD(cos(arg), -sin(arg));
1110 }
1111
1112 for(int i=0;i<fNumNus;i++){
1113 fBuffer[i] = 0;
1114 for(int j=0;j<fNumNus;j++){
1115 fBuffer[i] += conj(fEvec[j][i]) * fNuState[j];
1116 }
1117 fBuffer[i] *= fPhases[i];
1118 }
1119
1120 // Propagate neutrino state
1121 for(int i=0;i<fNumNus;i++){
1122 fNuState[i] = 0;
1123 for(int j=0;j<fNumNus;j++){
1124 fNuState[i] += fEvec[i][j] * fBuffer[j];
1125 }
1126 }
1127
1128}

1674vectorD PMNS_Base::ConvertEtoLoE(double E, double dE){

1677 // Use average if multiple paths
1678 SetCurPath(AvgPath(fNuPaths));

1689 LoEbin[0] = 0.5 * (fPath.length/(E-dE/2) + fPath.length/(E+dE/2));
1690 LoEbin[1] = fPath.length/(E-dE/2) - fPath.length/(E+dE/2);

1694 LoEbin[0] = 0.5 * (fPath.length/minE + fPath.length/(E+dE/2));
1695 LoEbin[1] = fPath.length/minE - fPath.length/(E+dE/2);

1805double PMNS_Base::AvgProbLoE(vectorC nu_in, int flvf, double LoE, double dLoE)
1806{
1807
1808 // Do nothing if L/E is not positive
1809 if(LoE<=0) return 0;
1810
1811 if(fNuPaths.empty()) return 0;
1812
1813 // Make sure fPath is set
1814 // Use average if multiple paths
1815 SetCurPath(AvgPath(fNuPaths));
1816
1817 // Set the energy at bin center
1818 SetEnergy(fPath.length/LoE);
1819
1820 // Don't average zero width
1821 if(dLoE<=0) return Prob(nu_in, flvf);
1822
1823 // Get sample points for this bin
1824 vectorD samples = GetSamplePoints(LoE, dLoE);
1825
1826 // Variables to fill sample
1827 // probabilities and weights
1828 double sumw = 0;
1829 double prob = 0;
1830 double length = fPath.length;
1831
1832 // Loop over all sample points
1833 for(int j=0; j<int(samples.size()); j++){
1834
1835 // Set (L/E)^-2 weights
1836 double w = 1./pow(samples[j],2);
1837
1838 // Add weighted probability
1839 prob += w * Prob(nu_in, flvf, length / samples[j]);
1840
1841 // Increment sum of weights
1842 sumw += w;
1843
1844 }
1845
1846 // Return weighted average of probabilities
1847 return prob / sumw;
1848
1849}

2163vectorD PMNS_Base::GetSamplePoints(double LoE, double dLoE)
2164{
2165
2166 // Solve Hamiltonian to get eigenvalues
2167 SolveHam();
2168
2169 // Define conversion factor [km/GeV -> 1/(4 eV^2)]
2170 const double k1267 = kKm2eV / (4 * kGeV2eV);
2171
2172 // Get list of all effective Dm^2
2173 vectorD effDm;
2174
2175 for(int i=0; i<fNumNus-1; i++){
2176 for(int j=i+1; j<fNumNus; j++){
2177 effDm.push_back( 2 * kGeV2eV * fEnergy * fabs(fEval[j] - fEval[i]) );
2178 }
2179 }
2180
2181 int numDm = effDm.size();
2182
2183 // Sort the effective Dm^2 list
2184 sort(effDm.begin(), effDm.end());
2185
2186 // Set a number of sub-divisions to achieve "good" accuracy
2187 // This needs to be studied better
2188 int n_div = ceil( 200 * pow(dLoE/LoE,0.8) / sqrt(fAvgProbPrec/1e-4) );
2189 //int n_div = 1;
2190
2191 // A vector to store sample points
2192 vectorD allSamples;
2193
2194 // Loop over sub-divisions
2195 for(int k=0; k<n_div; k++){
2196
2197 // Define sub-division center and width
2198 double bctr = LoE - dLoE/2 + (k+0.5)*dLoE/n_div;
2199 double bwdt = dLoE/n_div;
2200
2201 // Make a vector of L/E sample values
2202 // Initialized in the sub-division center
2203 vectorD samples;
2204 samples.push_back(bctr);
2205
2206 // Loop over all Dm^2 to average each frequency
2207 // This will recursively sample points in smaller
2208 // bins so that all relevant frequencies are used
2209 for(int i=0; i<numDm; i++){
2210
2211 // Copy the list of sample L/E values
2212 vectorD prev = samples;
2213
2214 // Redefine bin width to lie within full sub-division
2215 double Width = 2*min(prev[0] - (bctr - bwdt/2), (bctr + bwdt/2) - prev[0]);
2216
2217 // Compute oscillation argument sorted from lowest to highest
2218 const double arg = k1267 * effDm[i] * Width;
2219
2220 // Skip small oscillation values.
2221 // If it's the last one, lower the tolerance
2222 if(i < numDm-1){
2223 if(arg<0.9) continue;
2224 }
2225 else{
2226 if(arg<0.1) continue;
2227 }
2228
2229 // Reset samples to redefine them
2230 samples.clear();
2231
2232 // Loop over previous samples
2233 for(int j=0; j<int(prev.size()); j++){
2234
2235 // Compute new sample points around old samples
2236 // This is based on a oscillatory quadrature rule
2237 double sample = (1/sqrt(3)) * (Width/2);
2238 if(arg!=0) sample = acos(sin(arg)/arg)/arg * (Width/2);
2239
2240 // Add samples above and below center
2241 samples.push_back(prev[j]-sample);
2242 samples.push_back(prev[j]+sample);
2243
2244 }
2245
2246 }// End of loop over Dm^2
2247
2248 // Add sub-division samples to the end of allSamples vector
2249 allSamples.insert(allSamples.end(), samples.begin(), samples.end());
2250
2251 }// End of loop over sub-divisions
2252
2253 // Return all sample points
2254 return allSamples;
2255
2256}
