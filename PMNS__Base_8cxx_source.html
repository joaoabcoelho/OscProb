143bool PMNS_Base::TryCache()
144{
145
146 if(fUseCache && !fMixCache.empty()){
147
148 fProbe.SetVars(fEnergy, fPath, fIsNuBar);
149
150 unordered_set<EigenPoint>::iterator it = fMixCache.find(fProbe);
151
152 if(it != fMixCache.end()){
153 for(int i=0; i<fNumNus; i++){
154 fEval[i] = (*it).fEval[i] * (*it).fEnergy / fEnergy;
155 for(int j=0; j<fNumNus; j++){
156 fEvec[i][j] = (*it).fEvec[i][j];
157 }
158 }
159 return true;
160 }
161
162 }
163
164 return false;
165

499
500 // Get the sizes of the attribute and
501 // path sequence vectors
502 int nA = att.size();
503 int nP = fNuPaths.size();
504
505 // If the vector sizes are equal, update this attribute
506 if(nA == nP){
507
508 for(int i=0; i<nP; i++){
509
510 switch(idx){
511 case 0: fNuPaths[i].length = att[i]; break;
512 case 1: fNuPaths[i].density = att[i]; break;
513 case 2: fNuPaths[i].zoa = att[i]; break;
514 case 3: fNuPaths[i].layer = att[i]; break;
515 }
516
517 }
518
519 }
520 // If the vector sizes differ, create a new path sequence
521 // and set value for this attribute. Other attributes will
522 // be taken from default single path.
523 else{
524
525 cout << "Warning: New vector size. Starting new path vector." << endl;
526 cout << "To avoid possible issues, use the SetPath function." << endl;
527
528 // Start a new standard path just
529 // to set default values
530 SetStdPath();
531
532 // Create a path segment with default values
533 NuPath p = fNuPaths[0];
534
535 // Clear the path sequence
536 ClearPath();
537
538 // Set this particular attribute's value
539 // and add the path segment to the sequence
540 for(int i=0; i<nA; i++){
541
542 switch(idx){
543 case 0: p.length = att[i]; break;
544 case 1: p.density = att[i]; break;
545 case 2: p.zoa = att[i]; break;
546 case 3: p.layer = att[i]; break;

857//......................................................................
863void PMNS_Base::RotateState(int i, int j){
864
865 // Do nothing if angle is zero
866 if(fTheta[i][j]==0) return;
867
868 double sij = sin(fTheta[i][j]);
869 double cij = cos(fTheta[i][j]);
870
871 complexD buffer;
872
873 if(i+1==j || fDelta[i][j]==0){
874 buffer = cij*fNuState[i] + sij*fNuState[j];
875 fNuState[j] = cij*fNuState[j] - sij*fNuState[i];
876 }
877 else {
878 complexD eij = complexD(cos(fDelta[i][j]), -sin(fDelta[i][j]));
879 buffer = cij*fNuState[i] + sij*eij*fNuState[j];
880 fNuState[j] = cij*fNuState[j] - sij*conj(eij)*fNuState[i];
881 }
882
883 fNuState[i] = buffer;
884
885}

928void PMNS_Base::RotateH(int i,int j, matrixC& Ham){
929
930 // Do nothing if angle is zero
931 if(fTheta[i][j]==0) return;
932
933 double fSinBuffer = sin(fTheta[i][j]);
934 double fCosBuffer = cos(fTheta[i][j]);
935
936 double fHmsBufferD;
937 complexD fHmsBufferC;
938
939 // With Delta
940 if(i+1<j){
941 complexD fExpBuffer = complexD(cos(fDelta[i][j]), -sin(fDelta[i][j]));
942
943 // General case
944 if(i>0){
945 // Top columns
946 for(int k=0; k<i; k++){
947 fHmsBufferC = Ham[k][i];
948
949 Ham[k][i] *= fCosBuffer;
950 Ham[k][i] += Ham[k][j] * fSinBuffer * conj(fExpBuffer);
951
952 Ham[k][j] *= fCosBuffer;
953 Ham[k][j] -= fHmsBufferC * fSinBuffer * fExpBuffer;
954 }
955
956 // Middle row and column
957 for(int k=i+1; k<j; k++){
958 fHmsBufferC = Ham[k][j];
959
960 Ham[k][j] *= fCosBuffer;
961 Ham[k][j] -= conj(Ham[i][k]) * fSinBuffer * fExpBuffer;
962
963 Ham[i][k] *= fCosBuffer;
964 Ham[i][k] += fSinBuffer * fExpBuffer * conj(fHmsBufferC);
965 }
966
967 // Nodes ij
968 fHmsBufferC = Ham[i][i];
969 fHmsBufferD = real(Ham[j][j]);
970
971 Ham[i][i] *= fCosBuffer * fCosBuffer;
972 Ham[i][i] += 2 * fSinBuffer * fCosBuffer * real(Ham[i][j] * conj(fExpBuffer));
973 Ham[i][i] += fSinBuffer * Ham[j][j] * fSinBuffer;
974
975 Ham[j][j] *= fCosBuffer * fCosBuffer;
976 Ham[j][j] += fSinBuffer * fHmsBufferC * fSinBuffer;
977 Ham[j][j] -= 2 * fSinBuffer * fCosBuffer * real(Ham[i][j] * conj(fExpBuffer));
978
979 Ham[i][j] -= 2 * fSinBuffer * real(Ham[i][j] * conj(fExpBuffer)) * fSinBuffer * fExpBuffer;
980 Ham[i][j] += fSinBuffer * fCosBuffer * (fHmsBufferD - fHmsBufferC) * fExpBuffer;
981
982 }
983 // First rotation on j (No top columns)
984 else{
985 // Middle rows and columns
986 for(int k=i+1; k<j; k++){
987 Ham[k][j] = -conj(Ham[i][k]) * fSinBuffer * fExpBuffer;
988
989 Ham[i][k] *= fCosBuffer;
990 }
991
992 // Nodes ij
993 fHmsBufferD = real(Ham[i][i]);
994
995 Ham[i][j] = fSinBuffer * fCosBuffer * (Ham[j][j] - fHmsBufferD) * fExpBuffer;
996
997 Ham[i][i] *= fCosBuffer * fCosBuffer;
998 Ham[i][i] += fSinBuffer * Ham[j][j] * fSinBuffer;
999
1000 Ham[j][j] *= fCosBuffer * fCosBuffer;
1001 Ham[j][j] += fSinBuffer * fHmsBufferD * fSinBuffer;
1002 }
1003
1004 }
1005 // Without Delta (No middle rows or columns: j = i+1)
1006 else{
1007 // General case
1008 if(i>0){
1009 // Top columns
1010 for(int k=0; k<i; k++){
1011 fHmsBufferC = Ham[k][i];
1012
1013 Ham[k][i] *= fCosBuffer;
1014 Ham[k][i] += Ham[k][j] * fSinBuffer;
1015
1016 Ham[k][j] *= fCosBuffer;
1017 Ham[k][j] -= fHmsBufferC * fSinBuffer;
1018 }
1019
1020 // Nodes ij
1021 fHmsBufferC = Ham[i][i];
1022 fHmsBufferD = real(Ham[j][j]);
1023
1024 Ham[i][i] *= fCosBuffer * fCosBuffer;
1025 Ham[i][i] += 2 * fSinBuffer * fCosBuffer * real(Ham[i][j]);
1026 Ham[i][i] += fSinBuffer * Ham[j][j] * fSinBuffer;
1027
1028 Ham[j][j] *= fCosBuffer * fCosBuffer;
1029 Ham[j][j] += fSinBuffer * fHmsBufferC * fSinBuffer;
1030 Ham[j][j] -= 2 * fSinBuffer * fCosBuffer * real(Ham[i][j]);
1031
1032 Ham[i][j] -= 2 * fSinBuffer * real(Ham[i][j]) * fSinBuffer;
1033 Ham[i][j] += fSinBuffer * fCosBuffer * (fHmsBufferD - fHmsBufferC);
1034
1035 }
1036 // First rotation (theta12)
1037 else{
1038
1039 Ham[i][j] = fSinBuffer * fCosBuffer * Ham[j][j];
1040
1041 Ham[i][i] = fSinBuffer * Ham[j][j] * fSinBuffer;
1042
1043 Ham[j][j] *= fCosBuffer * fCosBuffer;
1044
1045 }
1046 }
1047
1048}

1062void PMNS_Base::BuildHms()
1063{
1064
1065 // Check if anything changed
1066 if(fBuiltHms) return;
1067
1068 // Tag to recompute eigensystem
1069 fGotES = false;
1070
1071 for(int j=0; j<fNumNus; j++){
1072 // Set mass splitting
1073 fHms[j][j] = fDm[j];
1074 // Reset off-diagonal elements
1075 for(int i=0; i<j; i++){
1076 fHms[i][j] = 0;
1077 }
1078 // Rotate j neutrinos
1079 for(int i=0; i<j; i++){
1080 RotateH(i,j,fHms);
1081 }
1082 }
1083
1084 ClearCache();
1085
1086 // Tag as built
1087 fBuiltHms = true;
1088

1096void PMNS_Base::PropagatePath(NuPath p)
1097{
1098
1099 // Set the neutrino path
1100 SetCurPath(p);
1101
1102 // Solve for eigensystem
1103 SolveHam();
1104
1105 double LengthIneV = kKm2eV * p.length;
1106 for(int i=0; i<fNumNus; i++){
1107 double arg = fEval[i] * LengthIneV;
1108 fPhases[i] = complexD(cos(arg), -sin(arg));
1109 }
1110
1111 for(int i=0;i<fNumNus;i++){
1112 fBuffer[i] = 0;
1113 for(int j=0;j<fNumNus;j++){
1114 fBuffer[i] += conj(fEvec[j][i]) * fNuState[j];
1115 }
1116 fBuffer[i] *= fPhases[i];
1117 }
1118
1119 // Propagate neutrino state
1120 for(int i=0;i<fNumNus;i++){
1121 fNuState[i] = 0;
1122 for(int j=0;j<fNumNus;j++){
1123 fNuState[i] += fEvec[i][j] * fBuffer[j];
1124 }
1125 }
1126
1127}

1648vectorD PMNS_Base::ConvertEtoLoE(double E, double dE){
1649

1651 // Use average if multiple paths
1652 SetCurPath(AvgPath(fNuPaths));

1654 // Define L/E variables
1655 vectorD LoEbin(2);
1656
1657 // Set a minimum energy
1658 double minE = 0.1 * E;
1659
1660 // Transform range to L/E
1661 // Full range if low edge > minE
1662 if(E-dE/2 > minE){
1663 LoEbin[0] = 0.5 * (fPath.length/(E-dE/2) + fPath.length/(E+dE/2));
1664 LoEbin[1] = fPath.length/(E-dE/2) - fPath.length/(E+dE/2);

1667 else{
1668 LoEbin[0] = 0.5 * (fPath.length/minE + fPath.length/(E+dE/2));
1669 LoEbin[1] = fPath.length/minE - fPath.length/(E+dE/2);
1670 }

1780{

1784
1785 if(fNuPaths.empty()) return 0;
1786

1788 // Use average if multiple paths
1789 SetCurPath(AvgPath(fNuPaths));
1790
1791 // Set the energy at bin center
1792 SetEnergy(fPath.length/LoE);
1793
1794 // Don't average zero width
1795 if(dLoE<=0) return Prob(nu_in, flvf);
1796
1797 // Get sample points for this bin
1798 vectorD samples = GetSamplePoints(LoE, dLoE);

2122vectorD PMNS_Base::GetSamplePoints(double LoE, double dLoE)
2123{
2124
2125 // Solve Hamiltonian to get eigenvalues
2126 SolveHam();
2127
2128 // Define conversion factor [km/GeV -> 1/(4 eV^2)]
2129 const double k1267 = kKm2eV / (4 * kGeV2eV);
2130
2131 // Get list of all effective Dm^2
2132 vectorD effDm;
2133
2134 for(int i=0; i<fNumNus-1; i++){
2135 for(int j=i+1; j<fNumNus; j++){
2136 effDm.push_back( 2 * kGeV2eV * fEnergy * fabs(fEval[j] - fEval[i]) );
2137 }
2138 }
2139
2140 int numDm = effDm.size();
2141
2142 // Sort the effective Dm^2 list
2143 sort(effDm.begin(), effDm.end());
2144
2145 // Set a number of sub-divisions to achieve "good" accuracy
2146 // This needs to be studied better
2147 int n_div = ceil( 20 * pow(dLoE/LoE,0.8) );
2148 //int n_div = 1;
2149
2150 // A vector to store sample points
2151 vectorD allSamples;
2152
2153 // Loop over sub-divisions
2154 for(int k=0; k<n_div; k++){
2155
2156 // Define sub-division center and width
2157 double bctr = LoE - dLoE/2 + (k+0.5)*dLoE/n_div;
2158 double bwdt = dLoE/n_div;
2159
2160 // Make a vector of L/E sample values
2161 // Initialized in the sub-division center
2162 vectorD samples;
2163 samples.push_back(bctr);
2164
2165 // Loop over all Dm^2 to average each frequency
2166 // This will recursively sample points in smaller
2167 // bins so that all relevant frequencies are used
2168 for(int i=0; i<numDm; i++){
2169
2170 // Copy the list of sample L/E values
2171 vectorD prev = samples;
2172
2173 // Redefine bin width to lie within full sub-division
2174 double Width = 2*min(prev[0] - (bctr - bwdt/2), (bctr + bwdt/2) - prev[0]);
2175
2176 // Compute oscillation argument sorted from lowest to highest
2177 const double arg = k1267 * effDm[i] * Width;
2178
2179 // Skip small oscillation values.
2180 // If it's the last one, lower the tolerance
2181 if(i < numDm-1){
2182 if(arg<0.9) continue;
2183 }
2184 else{
2185 if(arg<0.1) continue;
2186 }
2187
2188 // Reset samples to redefine them
2189 samples.clear();
2190
2191 // Loop over previous samples
2192 for(int j=0; j<int(prev.size()); j++){
2193
2194 // Compute new sample points around old samples
2195 // This is based on a oscillatory quadrature rule
2196 double sample = (1/sqrt(3)) * (Width/2);
2197 if(arg!=0) sample = acos(sin(arg)/arg)/arg * (Width/2);
2198
2199 // Add samples above and below center
2200 samples.push_back(prev[j]-sample);
2201 samples.push_back(prev[j]+sample);
2202
2203 }
2204
2205 }// End of loop over Dm^2
2206
2207 // Add sub-division samples to the end of allSamples vector
2208 allSamples.insert(allSamples.end(), samples.begin(), samples.end());
2209
2210 }// End of loop over sub-divisions
2211
2212 // Return all sample points
2213 return allSamples;
2214
2215}
