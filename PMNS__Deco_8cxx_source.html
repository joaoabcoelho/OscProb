88void PMNS_Deco::SetGamma32(double val){
89
90 if(val<0){
91 cout << "Warning: Gamma_32 must be positive. "
92 << "Setting it to absolute value of input: "
93 << -val << endl;
94 val = -val;
95 }
96
97 double min32 = 0.25*fGamma[1];
98
99 if(fGamma[0] >= 0) min32 *= 1 - pow(fGamma[0],2);
100 else min32 *= 1 + 3*pow(fGamma[0],2);
101
102 if(val < min32){
103
104 if(fGamma[0]>=0 || 4*val/fGamma[1] < 1){
105 fGamma[0] = sqrt(1 - 4*val/fGamma[1]);
106 fGamma[2] = 0.25*fGamma[1]*(1 + 3*pow(fGamma[0],2));
107 }
108 else{
109 fGamma[0] = -sqrt((4*val/fGamma[1] - 1)/3);
110 fGamma[2] = 0.25*fGamma[1]*(1 - pow(fGamma[0],2));
111 }
112
113 cout << "WARNING: Impossible to have Gamma32 = " << val
114 << " with current Gamma21 and theta parameters." << endl
115 << " Changing the value of cos(theta) to " << fGamma[0]
116 << endl;
117
118 return;
119
120 }
121
122 double arg = fGamma[1] * ( 4*val - fGamma[1]*(1 - pow(fGamma[0],2)) );
123
124 if(arg < 0){
125 arg = 0;
126 fGamma[0] = sqrt(1 - 4*val/fGamma[1]);
127 cout << "WARNING: Imaginary Gamma31 found. Changing the value of cos(theta) to "
128 << fGamma[0] << endl;
129 }
130
131 double gamma31 = val + fGamma[1]*pow(fGamma[0],2) + fGamma[0] * sqrt(arg);
132
133 fGamma[2] = gamma31;
134
135 if(fabs(val - GetGamma(3,2)) > 1e-6){
136 cout << "ERROR: Failed sanity check: GetGamma(3,2) = "
137 << GetGamma(3,2) << " != " << val << endl;
138 }
139
140}

210double PMNS_Deco::GetGamma(int i, int j)
211{
212
213 if(i < j){
214 cout << "First argument should be larger than second argument" << endl;
215 cout << "Setting reverse order (Gamma_" << j << i << "). " << endl;
216 int temp = i;
217 i = j;
218 j = temp;
219 }
220 if(i<1 || i>3 || i <= j || j < 1){
221 cout << "Gamma_" << i << j << " not valid for " << fNumNus;
222 cout << " neutrinos. Returning 0." << endl;
223 return 0;
224 }
225
226 if(j == 1){
227 return fGamma[i-1];
228 }
229 else {
230 // Combine Gamma31, Gamma21 and an angle (fGamma[0] = cos(th)) to make Gamma32
231 double arg = fGamma[1] * ( 4*fGamma[2] - fGamma[1]*(1 - pow(fGamma[0],2)) );
232
233 if(arg < 0) return fGamma[1] - 3*fGamma[2];
234
235 return fGamma[2] + fGamma[1]*pow(fGamma[0],2) - fGamma[0] * sqrt(arg);
236
237 }

319void PMNS_Deco::PropagatePath(NuPath p)
320{
321
322 // Set the neutrino path
323 SetCurPath(p);
324
325 // Solve for eigensystem
326 SolveHam();
327
328 // The code below is not pretty or optimised at all. Lots of matrix
329 // multiplications are going on which could be otpimised due to
330 // symmetry. The index sorting is also terrible. Needs improving.
331
332 // Store rotation matrices
333 matrix U = fEvec;
334 matrix Ut = CTransp(fEvec);
335
336 // Rotate to effective mass basis
337 fRho = Dot(Ut, Dot(fRho, U));
338
339 // Compute evolution matrix
340 matrix evolve(3, row(3,1));
341
342 // Some ugly way of matching gamma and dmsqr indices
343 vector<int> dm_idx = GetSortedIndices(fDm);
344 vector<int> ev_idx = GetSortedIndices(fEval);
345
346 int idx[3];
347 for(int i=0; i<3; i++) idx[ev_idx[i]] = dm_idx[i];
348
349 double energyCorr = pow(fEnergy, fPower);
350
351 for(int j=0; j<3; j++){
352 for(int i=0; i<j; i++){
353 double gamma_ij = GetGamma(max(idx[i],idx[j])+1, min(idx[i],idx[j])+1) * energyCorr;
354 complexD arg = complexD(gamma_ij * kGeV2eV, fEval[i] - fEval[j]) * kKm2eV * p.length;
355 evolve[i][j] = exp(-arg);
356 evolve[j][i] = exp(-conj(arg));
357 }}
358
359 // Evolve density matrix
360 fRho = Mult(fRho, evolve);
361
362 // Rotate back to flavour basis
363 fRho = Dot(U, Dot(fRho, Ut));
364
365}
