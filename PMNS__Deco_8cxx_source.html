86void PMNS_Deco::SetGamma32(double val){
87
88 if(val<0){
89 cerr << "WARNING: Gamma_32 must be positive. "
90 << "Setting it to absolute value of input: "
91 << -val << endl;
92 val = -val;
93 }
94
95 double min32 = 0.25*fGamma[1];
96
97 if(fGamma[0] >= 0) min32 *= 1 - pow(fGamma[0],2);
98 else min32 *= 1 + 3*pow(fGamma[0],2);
99
100 if(val < min32){
101
102 if(fGamma[0]>=0 || 4*val/fGamma[1] < 1){
103 fGamma[0] = sqrt(1 - 4*val/fGamma[1]);
104 fGamma[2] = 0.25*fGamma[1]*(1 + 3*pow(fGamma[0],2));
105 }
106 else{
107 fGamma[0] = -sqrt((4*val/fGamma[1] - 1)/3);
108 fGamma[2] = 0.25*fGamma[1]*(1 - pow(fGamma[0],2));
109 }
110
111 cerr << "WARNING: Impossible to have Gamma32 = " << val
112 << " with current Gamma21 and theta parameters." << endl
113 << " Changing the value of cos(theta) to " << fGamma[0]
114 << endl;
115
116 return;
117
118 }
119
120 double arg = fGamma[1] * ( 4*val - fGamma[1]*(1 - pow(fGamma[0],2)) );
121
122 if(arg < 0){
123 arg = 0;
124 fGamma[0] = sqrt(1 - 4*val/fGamma[1]);
125 cerr << "WARNING: Imaginary Gamma31 found. Changing the value of cos(theta) to "
126 << fGamma[0] << endl;
127 }
128
129 double gamma31 = val + fGamma[1]*pow(fGamma[0],2) + fGamma[0] * sqrt(arg);
130
131 fGamma[2] = gamma31;
132
133 if(fabs(val - GetGamma(3,2)) > 1e-6){
134 cerr << "ERROR: Failed sanity check: GetGamma(3,2) = "
135 << GetGamma(3,2) << " != " << val << endl;
136 }
137
138}

208double PMNS_Deco::GetGamma(int i, int j)
209{
210
211 if(i < j){
212 cerr << "WARNING: First argument should be larger than second argument" << endl
213 << "Setting reverse order (Gamma_" << j << i << "). " << endl;
214 int temp = i;
215 i = j;
216 j = temp;
217 }
218 if(i<1 || i>3 || i <= j || j < 1){
219 cerr << "WARNING: Gamma_" << i << j << " not valid for " << fNumNus
220 << " neutrinos. Returning 0." << endl;
221 return 0;
222 }
223
224 if(j == 1){
225 return fGamma[i-1];
226 }
227 else {
228 // Combine Gamma31, Gamma21 and an angle (fGamma[0] = cos(th)) to make Gamma32
229 double arg = fGamma[1] * ( 4*fGamma[2] - fGamma[1]*(1 - pow(fGamma[0],2)) );
230
231 if(arg < 0) return fGamma[1] - 3*fGamma[2];
232
233 return fGamma[2] + fGamma[1]*pow(fGamma[0],2) - fGamma[0] * sqrt(arg);
234
235 }

245void PMNS_Deco::RotateState(bool to_mass)
246{
247
248 // buffer = rho . U
249 for(int i=0; i<fNumNus; i++){
250 for(int j=0; j<fNumNus; j++){
251 fMBuffer[i][j] = 0;
252 for(int k=0; k<fNumNus; k++){
253 if(to_mass) fMBuffer[i][j] += fRho[i][k] * fEvec[k][j];
254 else fMBuffer[i][j] += fRho[i][k] * conj(fEvec[j][k]);
255 }
256 }}
257
258 // rho = U^\dagger . buffer = U^\dagger . rho . U
259 // Final matrix is Hermitian, so copy upper to lower triangle
260 for(int i=0; i<fNumNus; i++){
261 for(int j=i; j<fNumNus; j++){
262 fRho[i][j] = 0;
263 for(int k=0; k<fNumNus; k++){
264 if(to_mass) fRho[i][j] += conj(fEvec[k][i]) * fMBuffer[k][j];
265 else fRho[i][j] += fEvec[i][k] * fMBuffer[k][j];
266 }
267 if(j>i) fRho[j][i] = conj(fRho[i][j]);
268 }}
269
270}

276vectorI sort3(const vectorD& x){
277
278 vectorI out(3, 0);
279
280 // 1st element is smallest
281 if(x[0]<x[1] && x[0]<x[2]){
282 // 3rd element is largest
283 if(x[1]<x[2]){
284 out[1] = 1;
285 out[2] = 2;
286 }
287 // 2nd element is largest
288 else {
289 out[1] = 2;
290 out[2] = 1;
291 }
292 }
293 // 2nd element is smallest
294 else if(x[1]<x[2]){
295 out[0] = 1;
296 // 3rd element is largest
297 if(x[0]<x[2]) out[2] = 2;
298 // 1st element is largest
299 else out[1] = 2;
300 }
301 // 3rd element is smallest
302 else {
303 out[0] = 2;
304 // 2nd element is largest
305 if(x[0]<x[1]) out[2] = 1;
306 // 1st element is largest
307 else out[1] = 1;
308 }
309
310 return out;
311
312}

320void PMNS_Deco::PropagatePath(NuPath p)
321{
322
323 // Set the neutrino path
324 SetCurPath(p);
325
326 // Solve for eigensystem
327 SolveHam();
328
329 // Rotate to effective mass basis
330 RotateState(true);
331
332 // Some ugly way of matching gamma and dmsqr indices
333 vectorI dm_idx = sort3(fDm);
334 vectorI ev_idx = sort3(fEval);
335
336 int idx[3];
337 for(int i=0; i<fNumNus; i++) idx[ev_idx[i]] = dm_idx[i];
338
339 // Power law dependency of gamma parameters
340 double energyCorr = pow(fEnergy, fPower);
341
342 // Convert path length to eV
343 double lengthIneV = kKm2eV * p.length;
344
345 // Apply phase rotation to off-diagonal elements
346 for(int j=0; j<fNumNus; j++){
347 for(int i=0; i<j; i++){
348 // Get the appropriate gamma parameters based on sorted indices
349 double gamma_ij = GetGamma(max(idx[i],idx[j])+1, min(idx[i],idx[j])+1) * energyCorr;
350 // Multiply by path length
351 gamma_ij *= kGeV2eV * lengthIneV;
352 // This is the standard oscillation phase
353 double arg = (fEval[i] - fEval[j]) * lengthIneV;
354 // apply phase to rho
355 fRho[i][j] *= exp(-gamma_ij) * complexD(cos(arg), -sin(arg));
356 fRho[j][i] = conj(fRho[i][j]);
357 }}
358
359 // Rotate back to flavour basis
360 RotateState(false);
361
362}

438matrixD PMNS_Deco::ProbMatrix(int nflvi, int nflvf)
439{
440
441 assert(nflvi<=fNumNus && nflvi>=0);
442 assert(nflvf<=fNumNus && nflvf>=0);
443
444 // Output probabilities
445 matrixD probs(nflvi, vectorD(nflvf));
446
447 // List of states
448 vector<matrixC> allstates(nflvi, matrixC(fNumNus, vectorC(fNumNus)));
449
450 // Reset all initial states
451 for(int i=0; i<nflvi; i++){
452 ResetToFlavour(i);
453 allstates[i] = fRho;
454 }
455
456 // Propagate all states in parallel
457 for(int i=0; i<int(fNuPaths.size()); i++){
458
459 for(int flvi=0; flvi<nflvi; flvi++){
460 fRho = allstates[flvi];
461 PropagatePath(fNuPaths[i]);
462 allstates[flvi] = fRho;
463 }
464
465 }
466
467 // Get all probabilities
468 for(int flvi=0; flvi<nflvi; flvi++){
469 for(int flvj=0; flvj<nflvf; flvj++){
470 probs[flvi][flvj] = abs(allstates[flvi][flvj][flvj]);
471 }}
472
473 return probs;
474
475}
