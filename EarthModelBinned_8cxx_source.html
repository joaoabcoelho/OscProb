26void TrajConstants::UpdateNuAngles(double cosTheta, double phi) {
27 cosT = cosTheta;
28 sinSqT = 1 - cosT*cosT;
29 sinT = sqrt(sinSqT);
30
31 cosA = cos(phi);
32 sinA = sin(phi);
33
34 sinTsinA = sinT*sinA;
35 sinTcosA = sinT*cosA;
36}

46void TrajConstants::UpdateDetPos(double rDet, double DetLat, double DetLon) {
47 sinDetLat = sin(DetLat);
48 cosDetLon = cos(DetLon);
49 sinDetLon = sin(DetLon);
50 DetRadius = rDet;
51
52 cosDetLat = 1 - sinDetLat*sinDetLat;
53 rDetSinDetLat = DetRadius*sinDetLat;
54 rDetCosDetLat = DetRadius*cosDetLat;
55}

62void TrajConstants::Recalculate() {
63 cosTcosDetLat = cosT*cosDetLat;
64 sinTsinAsinDetLon = sinTsinA*sinDetLon;
65 sinTsinAcosDetLon = sinTsinA*cosDetLon;
66 rDetCosT = DetRadius*cosT;
67 rDetSinT = DetRadius*sinT;
68 rDetCosAcosDetLat = cosA*rDetCosDetLat;
69 alpha = sinTcosA*sinDetLat-cosTcosDetLat;
70 beta = sinT*sinDetLat-cosA*cosTcosDetLat;
71 gamma = sinTcosA*cosDetLat+cosT*sinDetLat;
72 gammaSq = gamma*gamma;
73 rDetGammaSinDetLat = rDetSinDetLat*gamma;
74 maxSinSqLat = 1-pow(sinA*cosDetLat,2);
75}

183void EarthModelBinned::LoadModel(string filename)
184{
185
186 // Clear the current model
187 ClearModel();
188 double EarthRadius = 6371.0;
189
190 // Use default if no file provided
191 if(filename == ""){
192 filename = PREM3D_DEFAULT;
193 }
194 cout << "Loading Earth model table from " << filename << "..." << endl;
195
196 // Open the file
197 ifstream fin;
198 fin.open(filename.c_str());
199 if(!fin){
200 cout << "ERROR: File " << filename << " not found!" << endl;
201 return;
202 }
203
204 // Variables for storing table rows
205 float depth, latitude, longitude, density, zoa, index;
206
207 // Keep track of previous depth/latitude/longitude
208 double radius_prev = 0;
209 double lat_prev = -90.0;
210 double lon_prev = 0;
211
212 // Hold onto previous different depth, so we have depth maximum for bin
213 double radius_min = radius_prev;
214
215 // Loop over table rows
216 while(fin >> longitude >> latitude >> depth >> density >> zoa >> index){
217 double outer_radius = EarthRadius - depth;
218
219 // Move minimum depth (and reset latitude/longitude) if depth has changed from previous one
220 if(outer_radius > radius_prev){
221 radius_min = radius_prev;
222 lat_prev = -90.0;
223 lon_prev = 0.0;
224 fnDepthBins++;
225 } else if(outer_radius < radius_prev){ // Depths must be ordered in model file
226 cout << "ERROR: Depths are not sorted in decreasing order in the model file (or depth is greater than " << EarthRadius << " km)." << endl;
227 ClearModel();
228 return;
229 } else if(longitude > lon_prev){ // Reset latitude if longitude has changed from previous one
230 lat_prev = -90.0;
231 fnLonBins++;
232 } else if(longitude < lon_prev){ // Longitudes must be ordered in model file (after depths)
233 cout << "ERROR: Longitudes are not sorted in increasing order (after depths) in the model file." << endl;
234 ClearModel();
235 return;
236 } else if(latitude < lat_prev){ // Latitudes must be ordered in model file (after longitudes)
237 cout << "ERROR: Latitudes are not sorted in increasing order (after longitudes) in the model file." << endl;
238 ClearModel();
239 return;
240 }
241
242 // Add this bin to the model
243 AddBin(outer_radius, radius_min, latitude, longitude, density, zoa, index);
244
245 // Set previous coordinates for next bin
246 radius_prev = outer_radius;
247 lat_prev = latitude;
248 lon_prev = longitude;
249
250 }
251
252 //Set Outermost Radius
253 fRadiusMax = radius_prev;
254
255 //Modify number of longitude bins to how many there actually are (instead of how many times we changed the longitude bin without changing the depth bin = nLonBins-1 times per depth bin)
256 fnLonBins = fnLonBins/fnDepthBins + 1;
257 fInvLonBinWidth = 0.5*fnLonBins/M_PI; //in radians^(-1)
258 fHalfLonBinWidth = M_PI/fnLonBins; //in radians
259
260 //Calculate number of latitude bins
261 fnLatBins = fEarthBins.size()/(fnLonBins*fnDepthBins);
262 fInvLatBinWidth = fnLatBins/M_PI; //in radians^(-1)
263 fHalfLatBinWidth = 0.5*M_PI/fnLatBins; //in radians
264
265 //Error if first latitude bin center deviates from about half of the calculated latitude bin width
266 if(abs(fHalfLatBinWidth-(fEarthBins[0].latitude+0.5*M_PI))>fHalfLatBinWidth*0.1) { //Warn if first latitude bin center isn't about half of the latitude bin width
267 cout << "ERROR: Latitude of 1st bin center (" << fEarthBins[0].latitude << ") is not in the middle of the first bin, whose size should be " << 2*fHalfLatBinWidth << " rad, as calculated from having " << fnLatBins << " latitude bins." << endl;
268 ClearModel();
269 return;
270 }
271
272 cout << "\t...done (" << fEarthBins.size() << " bins: " << fnDepthBins << " depth, " << fnLatBins << " latitude, and " << fnLonBins << " longitude)." << endl;
273}

386double EarthModelBinned::DetDistForNextLatBin(int cur_index, LatBinInfo &L)
387{
388
389 //Find next lat bin boundary
390 double NewLat = fEarthBins[cur_index].latitude + L.sign*L.dLat;
391 //Turn around if lat reaches +/- 90 degrees
392 if (abs(NewLat)-0.5*M_PI > -1.0e-14) {
393 if (L.maxreached > 0)
394 return -1; //Only turn around once
395 L.maxreached = 1;
396 L.sign = -L.sign;
397 NewLat = fEarthBins[cur_index].latitude + L.sign*L.dLat;
398 if (abs(NewLat)-0.5*M_PI > -1.0e-14)
399 return -1; //This would happen if there's only one latitude bin...
400 }
401
402 //Calculate parts of answer
403 double sinNewLat = sin(NewLat);
404 double NsinSqNewLat = -sinNewLat*sinNewLat;
405 double radicand = fC.maxSinSqLat + NsinSqNewLat;
406 //Turn around if the radicand is 0 (close enough) or less
407 if (radicand < 1.0e-14) {
408 if (L.maxreached > 0)
409 return -1; //Only turn around once
410 L.maxreached = 1;
411 L.sign = -L.sign;
412 NewLat = fEarthBins[cur_index].latitude + L.sign*L.dLat;
413 sinNewLat = sin(NewLat);
414 NsinSqNewLat = -sinNewLat*sinNewLat;
415 radicand = fC.maxSinSqLat + NsinSqNewLat;
416 if (radicand < 1.0e-14)
417 return -1; //The rest of the track is contained in this lon bin
418 }
419 double denominator = fC.gammaSq + NsinSqNewLat;
420
421 //Calculate distance from the detector at next lat bin boundary
422 double answer = 0;
423 if (denominator == 0) {
424 answer = 0.5*(fC.rDetCosAcosDetLat/fC.beta-fC.rDetSinDetLat/fC.gamma);
425 } else {
426 answer = -(fC.rDetCosT*NsinSqNewLat+fC.rDetGammaSinDetLat+L.sign*fC.rDetSinT*sinNewLat*sqrt(radicand))/denominator;
427 }
428 //Indicates that going to the edge of the bin jumped to an invalid part of the function
429 if (L.maxreached > 0 && answer >= fC.rDetCosAcosDetLat/fC.beta) {
430 return -1; //The rest of the track is contained in this lon bin
431 }
432
433 return answer;
434
435}

450double EarthModelBinned::DetDistForNextLonBin(double prev_lon, LonBinInfo &L)
451{
452
453 //Increment to next bin in direction dLon
454 double lon = prev_lon + L.dLon;
455 if (lon < 0 || lon >= 2*M_PI)
456 lon -= floor(lon/(2*M_PI));
457 if (L.dLon > 0) {
458 L.bin++;
459 if(L.bin >= fnLonBins) {
460 //cross from 2pi to 0
461 L.bin = 0;
462 }
463 } else {
464 L.bin--;
465 if(L.bin < 0) {
466 //cross from 0 to 2pi
467 L.bin = fnLonBins - 1;
468 }
469 }
470
471 //Check if lon is outside of range covered by neutrino's trajectory
472 if (L.max < L.min) {
473 if (L.min > lon && L.max < lon)
474 return -1;
475 } else {
476 if (L.min > lon || L.max < lon)
477 return -1;
478 }
479
480 //Parts of the answer
481 double cosNewLon = cos(lon);
482 double sinNewLon = sin(lon);
483 double expression = fC.sinDetLon * cosNewLon - sinNewLon * fC.cosDetLon;
484 double denominator = fC.alpha*expression + fC.sinTsinAcosDetLon*cosNewLon + fC.sinTsinAsinDetLon*sinNewLon;
485
486 //Calculate & return detector distance
487 if (denominator == 0) { //Is there ever a case when this is true???
488 //Error message will be printed at the end of the FillPath function
489 L.err_message.assign("Longitude along neutrino trajectory = "+std::to_string(lon));
490 L.error = -1;
491 return -1;
492 }
493 return fC.rDetCosDetLat * expression / denominator;
494
495}

511void EarthModelBinned::RecordLatLonBinCrossings(double detDist_nextDbin, double &DetDist, int &index, LatBinInfo &latI, LonBinInfo &lonI)
512{
513
514 while(detDist_nextDbin < latI.detDist_nextBin || detDist_nextDbin < lonI.detDist_nextBin) {
515
516 if (latI.detDist_nextBin > lonI.detDist_nextBin) {
517// cout << "\tChanging lat bins (to " << latI.nextBin << ") at x=" << latI.detDist_nextBin << "km" << endl;
518
519 //Add segment to next lat bin to path
520 AddPath(DetDist - latI.detDist_nextBin, fEarthBins[index]);
521
522 //Move lat bins
523 DetDist = latI.detDist_nextBin;
524 index += latI.nextBin - latI.bin;
525 latI.bin = latI.nextBin;
526
527 //Find next lat bin
528 if (latI.dLat == 0) {
529 latI.detDist_nextBin = -1;
530 } else {
531 latI.detDist_nextBin = DetDistForNextLatBin(index, latI);
532 latI.nextBin = latI.bin + latI.sign;
533 }
534 } else {
535// cout << "\tChanging lon bins (to " << lonI.nextBin << ") at x=" << lonI.detDist_nextBin << "km" << endl;
536
537 //Add segment to next lon bin to path
538 AddPath(DetDist - lonI.detDist_nextBin, fEarthBins[index]);
539
540 //Move lon bins
541 DetDist = lonI.detDist_nextBin;
542 index += (lonI.nextBin - lonI.bin)*fnLatBins;
543 lonI.bin = lonI.nextBin;
544
545 //Find next lon bin
546 if (lonI.dLon == 0) {
547 lonI.detDist_nextBin = -1;
548 } else {
549 lonI.detDist_nextBin = DetDistForNextLonBin(fEarthBins[index].longitude, lonI);
550 }
551 }
552
553 }

563int EarthModelBinned::LonBinIndex(double longitude)
564{
565
566 //Find difference between longitude and beginning of first lon bin
567 double londiff = longitude - (fEarthBins[0].longitude-fHalfLonBinWidth);
568
569 //Shift londiff to be between 0 and 2pi
570 if(londiff < 0 || londiff >= 2*M_PI)
571 londiff -= floor(londiff/(2*M_PI));
572
573 //Number of binwidths after from first lon bin
574 return floor(londiff*fInvLonBinWidth);

596int EarthModelBinned::FillPath(double cosT, double phi)
597{
598
599 // Clear current path sequence
600 fNuPath.clear();
601
602 // Do nothing if cosine is unphysical
603 if(fabs(cosT) > 1) return 0;
604
605 // Calculate useful combinations of variables
606 double Az = phi/180.0*M_PI; //convert from degrees to radians
607 fC.UpdateNuAngles(cosT, Az);
608 fC.Recalculate();
609 double rDetSqSinSqT = pow(fC.rDetSinT,2);
610 double distEdgeToMin = sqrt(fRadiusMax*fRadiusMax-rDetSqSinSqT); //distance from point where neutrino enters to minimum depth (neglecting stopping at the detector)
611 double baseline = distEdgeToMin - fC.rDetCosT; //total baseline
612
613 // Define the maximum depth (minimum radius) along the path (detector depth/radius, if cosT is non-negative)
614 double minRadius = fDetRadius;
615 if(cosT < 0) {minRadius = fC.rDetSinT;}
616
617 // Starting latitude
618 double init_lat = asin((fC.rDetSinT*fC.beta+fC.gamma*distEdgeToMin)/fRadiusMax); //in radians
619
620 // Starting longitude
621 double init_lon = atan2(fC.sinDetLon*fC.rDetCosDetLat-baseline*(fC.alpha*fC.sinDetLon+fC.sinTsinAcosDetLon), fC.cosDetLon*fC.rDetCosDetLat+baseline*(fC.sinTsinAsinDetLon-fC.alpha*fC.cosDetLon)); //in radians
622 double max_lon = init_lon+M_PI;
623 if (init_lon < 0) {
624 init_lon += 2*M_PI;
625 }
626 double min_lon = init_lon;
627
628 // Get initial Earth bin and # of bins per depth layer
629 int binsPerDepth = fEarthBins.size()/fnDepthBins;
630 int init_lonBin = LonBinIndex(init_lon);
631 int init_latBin = floor((init_lat+M_PI/2)*fInvLatBinWidth);
632 int index = fEarthBins.size()-binsPerDepth + init_lonBin*fnLatBins + init_latBin;
633// cout << "\t...Initial (Lat,Long) = (" << init_lat << "," << init_lon << ") => Initial bin: " << index << "..." << endl;
634
635 /* Find detector distances for 1st latitude/longitude bin changes */
636 LatBinInfo latinfo; //***Is this the right way to declare it?
637 latinfo.nextBin = init_latBin;
638 latinfo.dLat = fHalfLatBinWidth; //change in lat from bin center to next bin (0 if no more than 1 lat bin change)
639 LonBinInfo loninfo; //***Is this the right way to declare it?
640 loninfo.nextBin = init_lonBin;
641 //Condition calculation for starting sign in x(lat) equation
642 latinfo.sign = 1; //+ => lat incr., - => lat decr. (with decr. x)
643 latinfo.maxreached = 0; //this changes to 1 if past the lat func transition
644 if (fC.beta < 0)
645 latinfo.sign = -1;
646 if (fC.beta == 0) { if (fC.cosA > 0) latinfo.sign = -1;
647 } else if (baseline < fC.rDetCosAcosDetLat/fC.beta) {
648 latinfo.sign = -(latinfo.sign);
649 latinfo.maxreached++;
650 }
651 //Condition calculation for incr./decr. Longitude (with decr. x)
652 loninfo.dLon = fHalfLonBinWidth; //change in lon from bin center to next bin (includes direction; 0 if no more than 1 lon bin change)
653 if (fC.sinA < 0) {
654 loninfo.dLon = -(loninfo.dLon);
655 loninfo.min = loninfo.max;
656 loninfo.max = init_lon;
657 }
658 //Remainder of detDist calculations
659 if (fC.sinT == 0) {
660 //both latitude and logitude flip at center of the Earth
661 loninfo.detDist_nextBin = -fC.rDetCosT; //formula is divided by cosT, but cosT = +-1
662 loninfo.nextBin = LonBinIndex(fDetLon);
663 loninfo.dLon = 0;
664 latinfo.detDist_nextBin = loninfo.detDist_nextBin;
665 latinfo.nextBin = floor((fDetLat+0.5*M_PI)*fInvLatBinWidth);
666 latinfo.dLat = 0;
667 } else {
668 if (fC.sinA == 0) {
669 if (fC.alpha == 0) { //crosses "at" infinity
670 loninfo.detDist_nextBin = -1;
671 } else { //longitude flips at center of the Earth
672 loninfo.detDist_nextBin = fC.rDetCosDetLat/fC.alpha;
673 loninfo.nextBin = LonBinIndex(fDetLon);
674 }
675 loninfo.dLon = 0;
676 } else if (fC.cosDetLat == 0) {
677 loninfo.detDist_nextBin = -1;
678 } else {
679 //x(lon) calculation
680 loninfo.detDist_nextBin = DetDistForNextLonBin(fEarthBins[index].longitude, loninfo);
681 }
682 //x(lat) calculation
683 latinfo.detDist_nextBin = DetDistForNextLatBin(index, latinfo);
684 latinfo.nextBin = init_latBin + latinfo.sign;
685 }
686
687 // Get initial Detector Distance
688 double prev_DetDistance = baseline;
689
690 /* Start at the top layer of Earth model and go down to minimum (not including minimum), looping over depth bins */
691 int dBin = 0; //depth bin index
692 latinfo.bin = init_latBin;
693 loninfo.bin = init_lonBin;
694 for(dBin = 0; dBin<fnDepthBins-1; dBin++) {
695 //Check if minimum radius is contained in depth bin
696 double r_in = fEarthBins[index].radius_in; //inner-most radius
697 if(r_in < minRadius) break;
698
699 //Distance from detector at inner-most radius
700 double DetDistance = -fC.rDetCosT + sqrt(r_in*r_in - rDetSqSinSqT);
701
702 //Check for latitude/longitude bin crossings
703 RecordLatLonBinCrossings(DetDistance, prev_DetDistance, index, latinfo, loninfo);
704
705 //Add Segment to next depth bin to path
706 AddPath(prev_DetDistance - DetDistance, fEarthBins[index]);
707
708 //Reset variables for next depth bin
709 prev_DetDistance = DetDistance;
710 index -= binsPerDepth;
711 }
712
713 /* Do minimum depth bin and then go up to the detector */
714 int index2 = index;
715 for(index2 = index; fEarthBins[index2].radius_out < fDetRadius; index2+=binsPerDepth) {
716 double r_out = fEarthBins[index2].radius_out; //outer-most radius
717
718 //Distance from detector at outer-most radius
719 double DetDistance = -fC.rDetCosT - sqrt(r_out*r_out - rDetSqSinSqT);
720
721 //Check for latitude/longitude bin crossings
722 RecordLatLonBinCrossings(DetDistance, prev_DetDistance, index2, latinfo, loninfo);
723
724 //Add Segment in depth bin to path
725 AddPath(prev_DetDistance - DetDistance, fEarthBins[index2]);
726
727 //Reset variables for "previous" depth bin
728 prev_DetDistance = DetDistance;
729 }
730
731 /* Do path to detector */
732 //Check for latitude/longitude bin crossings
733 RecordLatLonBinCrossings(0, prev_DetDistance, index2, latinfo, loninfo);
734 //Add the path segment within the detector bin
735 AddPath(prev_DetDistance, fEarthBins[index2]);
736
737 //Longitude calculation error message, if needed
738 if (loninfo.error < 0) {
739 cerr << "ERROR: Oops... It turns out that I was wrong about the denominator of the x(long) equation never being 0 apart from the cases where sin(Zenith) = 0, sin(Azimuthal) = 0, or cos(Detector Longitude) = 0. Send the following message to rpestes@apc.in2p3.fr:" << endl << endl;
740 cerr << "You were wrong about the denominator of x(long)... Here are the values of the variables that broke it:" << endl;
741 cerr << "\tDetector Coordinates (r,lat,lon) = (" << fDetRadius << ", " << fDetLat << ", " << fDetLon << ")" << endl;
742 cerr << "\tcos(Zenith Angle) = " << cosT << endl;
743 cerr << "\tAzimuthal Angle = " << phi << " deg" << endl;
744 cerr << "\t" << loninfo.err_message << endl;
745 cerr << "Please fix this ASAP!" << endl;
746 return -1;
747 }
748
749 // Return the number of path segments
750 return fNuPath.size();
751
752}
