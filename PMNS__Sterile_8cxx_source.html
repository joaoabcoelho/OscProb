48void PMNS_Sterile::UpdateHam(){
49
50 double rho = fPath.density;
51 double zoa = fPath.zoa;
52
53 double lv = 2 * kGeV2eV*fEnergy; // 2E in eV
54 double kr2GNe = kK2*M_SQRT2*kGf * rho*zoa; // Electron matter potential in eV
55 double kr2GNn = kK2*M_SQRT2*kGf * rho*(1-zoa)/2; // Neutron matter potential in eV
56
57 // Finish building Hamiltonian in matter with dimension of eV
58 for(int i=0; i<fNumNus; i++){
59
60 // Set diagonal elements which are real
61 fHam(i,i) = fHms[i][i]/lv;
62
63 // Set off-diagonal elements and flip deltaCP if needed
64 for(int j=i+1; j<fNumNus; j++){
65 // Obs: fHam is lower tirangular while fHms is upper triangular
66 if(!fIsNuBar) fHam(j,i) = fHms[i][j]/lv;
67 else fHam(j,i) = conj(fHms[i][j])/lv;
68 }
69
70 // Subtract NC coherent forward scattering from sterile neutrinos.
71 // See arXiv:hep-ph/0606054v3, eq. 3.15, for example.
72 if(i>2){
73 if(!fIsNuBar) fHam(i,i) += kr2GNn;
74 else fHam(i,i) -= kr2GNn;
75 }
76
77 }
78
79 // Add nue CC coherent forward scattering.
80 if(!fIsNuBar) fHam(0,0) += kr2GNe;
81 else fHam(0,0) -= kr2GNe;
82
83}

98void PMNS_Sterile::SolveHam()
99{
100
101 // Build Hamiltonian
102 BuildHms();
103
104 // Check if anything changed
105 if(fGotES) return;
106
107 // Try caching if activated
108 if(TryCache()) return;
109
110 UpdateHam();
111
112 // Solve Hamiltonian for eigensystem
113 Eigen::SelfAdjointEigenSolver<Eigen::MatrixXcd> eigensolver(fHam);
114
115 // Fill fEval and fEvec vectors from GSL objects
116 for(int i=0; i<fNumNus; i++){
117 fEval[i] = eigensolver.eigenvalues()(i);
118 for(int j=0; j<fNumNus; j++){
119 fEvec[i][j] = eigensolver.eigenvectors()(i,j);
120 }
121 }
122
123 fGotES = true;
124
125 // Fill cache if activated
126 FillCache();
127
128}
