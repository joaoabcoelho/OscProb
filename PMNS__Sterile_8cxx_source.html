150void PMNS_Sterile::SolveHam()
151{
152
153 // Build Hamiltonian
154 BuildHms();
155
156 // Check if anything changed
157 if(fGotES) return;
158
159 // Try caching if activated
160 if(TryCache()) return;
161
162 double rho = fPath.density;
163 double zoa = fPath.zoa;
164
165 double lv = 2 * kGeV2eV*fEnergy; // 2E in eV
166 double kr2GNe = kK2*M_SQRT2*kGf * rho*zoa; // Electron matter potential in eV
167 double kr2GNn = kK2*M_SQRT2*kGf * rho*(1-zoa)/2; // Neutron matter potential in eV
168
169 // Finish building Hamiltonian in matter with dimension of eV
170 for(size_t i=0;i<size_t(fNumNus);i++){
171 complexD buf = fHms[i][i]/lv;
172 *gsl_matrix_complex_ptr(fGSL.H_GSL, i, i) = gsl_complex_rect(real(buf),0);
173 for(size_t j=i+1;j<size_t(fNumNus);j++){
174 buf = fHms[i][j]/lv;
175 if(!fIsNuBar) *gsl_matrix_complex_ptr(fGSL.H_GSL, j, i) = gsl_complex_rect(real(buf),-imag(buf));
176 else *gsl_matrix_complex_ptr(fGSL.H_GSL, j, i) = gsl_complex_rect(real(buf), imag(buf));
177 }
178 if(i>2){
179 // Subtract NC coherent forward scattering from sterile neutrinos. See arXiv:hep-ph/0606054v3, eq. 3.15, for example.
180 if(!fIsNuBar) *gsl_matrix_complex_ptr(fGSL.H_GSL, i, i) = gsl_complex_add_real(gsl_matrix_complex_get(fGSL.H_GSL,i,i) , kr2GNn);
181 else *gsl_matrix_complex_ptr(fGSL.H_GSL, i, i) = gsl_complex_add_real(gsl_matrix_complex_get(fGSL.H_GSL,i,i) , -kr2GNn);;
182 }
183 }
184 // Add nue CC coherent forward scattering from sterile neutrinos.
185 if(!fIsNuBar) *gsl_matrix_complex_ptr(fGSL.H_GSL, 0, 0) = gsl_complex_add_real(gsl_matrix_complex_get(fGSL.H_GSL,0,0) , kr2GNe);
186 else *gsl_matrix_complex_ptr(fGSL.H_GSL, 0, 0) = gsl_complex_add_real(gsl_matrix_complex_get(fGSL.H_GSL,0,0) , -kr2GNe);
187
188 // Solve Hamiltonian for eigensystem
189 gsl_eigen_hermv(fGSL.H_GSL, fGSL.fEvalGSL, fGSL.fEvecGSL, fGSL.W_GSL);
190
191 // Fill fEval and fEvec vectors from GSL objects
192 for(int i=0;i<fNumNus;i++){
193 fEval[i] = gsl_vector_get(fGSL.fEvalGSL,i);
194 for(int j=0;j<fNumNus;j++){
195 gsl_complex buf = gsl_matrix_complex_get(fGSL.fEvecGSL,i,j);
196 fEvec[i][j] = complexD( GSL_REAL(buf), GSL_IMAG(buf) );
197 }
198 }
199
200 fGotES = true;
201
202 // Fill cache if activated
203 FillCache();
204
205}
