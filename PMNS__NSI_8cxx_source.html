92void PMNS_NSI::SetEps(int flvi, int flvj, double val, double phase){
93
94 if(flvi > flvj){
95 cout << "First argument should be smaller or equal to second argument" << endl;
96 cout << "Setting reverse order (Eps_" << flvj << flvi << "). " << endl;
97 int temp = flvi;
98 flvi = flvj;
99 flvj = temp;
100 }
101 if(flvi<0 || flvi>2 || flvj < flvi || flvj > 2){
102 cout << "Eps_" << flvi << flvj << " not valid for " << fNumNus;
103 cout << " neutrinos. Doing nothing." << endl;
104 return;
105 }
106
107 complexD h = val;
108
109 if(flvi != flvj) h *= complexD(cos(phase), sin(phase));
110
111 bool isSame = (fEps[flvi][flvj] == h);
112
113 if(!isSame) ClearCache();
114
115 fGotES *= isSame;
116
117 fEps[flvi][flvj] = h;
118
119}

230void PMNS_NSI::UpdateHam()
231{
232
233 double lv = 2 * kGeV2eV*fEnergy; // 2*E in eV
234
235 double kr2GNe = kK2*M_SQRT2*kGf * fPath.density;
236 double kr2GNnsi = kr2GNe;
237
238 kr2GNe *= fPath.zoa; // Std matter potential in eV
239 kr2GNnsi *= GetZoACoup(); // NSI matter potential in eV
240
241 // Finish build Hamiltonian in matter with dimension of eV
242 for(int i=0;i<fNumNus;i++){
243 for(int j=i;j<fNumNus;j++){
244 if(!fIsNuBar) fHam[i][j] = fHms[i][j]/lv + kr2GNnsi*fEps[i][j];
245 else fHam[i][j] = conj(fHms[i][j]/lv - kr2GNnsi*fEps[i][j]);
246 }
247 }
248
249 if(!fIsNuBar) fHam[0][0] += kr2GNe;
250 else fHam[0][0] -= kr2GNe;
251
252}

346double PMNS_NSI::GetZoACoup()
347{
348
349 return fNSIcoup[0] * fPath.zoa // electrons: Z
350 + fNSIcoup[1] * (1 + fPath.zoa) // u-quarks: A + Z
351 + fNSIcoup[2] * (2 - fPath.zoa); // d-quarks: 2A - Z
352
