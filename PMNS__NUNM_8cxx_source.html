96void PMNS_NUNM::SetAlpha(int i, int j, double val, double phase)
97{
98 if (i < j) {
99 cerr << "WARNING: First argument should be larger or equal to second "
100 "argument"
101 << endl
102 << "Setting reverse order (Alpha_" << j << i << "). " << endl;
103 int temp = i;
104 i = j;
105 j = temp;
106 }
107 if (i < 0 || i > 2 || j > i || j > 2) {
108 cerr << "WARNING: Alpha_" << i << j << " not valid for " << fNumNus
109 << " neutrinos. Doing nothing." << endl;
110 return;
111 }
112
113 complexD h = val;
114
115 if (i == j) { h = 1. + val; }
116 else {
117 h *= complexD(cos(phase), sin(phase));
118 }
119
120 bool isSame = (Alpha(i, j) == h);
121
122 if (!isSame) ClearCache();
123
124 fGotES *= isSame;
125
126 Alpha(i, j) = h;
127}

257void PMNS_NUNM::PropagatePath(NuPath p)
258{
259 // Set the neutrino path
260 SetCurPath(p);
261 X = Alpha * Alpha.adjoint();
262 if (fscale == 1) { // normalise mixing matrix in high scale scenario to ensure
263 // completeness
264 for (int i = 0; i < 3; ++i) {
265 for (int j = 0; j < i + 1; ++j) {
266 Alpha(i, j) *=
267 1 / std::sqrt(X(i, i).real()); // Scale by the inverse square root
268 // of the diagonal elements of X
269 }
270 }
271 }
272
273 // Solve for eigensystem
274 SolveHam();
275
276 for (int i = 0; i < 3; ++i) {
277 for (int j = 0; j < 3; ++j) { Evec0(i, j) = fEvec[i][j]; }
278 }
279
280 Evec = Alpha * Evec0;
281 EvecA = Evec.inverse();
282
283 double LengthIneV = kKm2eV * p.length;
284 for (int i = 0; i < fNumNus; i++) {
285 double arg = fEval[i] * LengthIneV;
286 fPhases[i] = complexD(cos(arg), -sin(arg));
287 }
288
289 for (int i = 0; i < fNumNus; i++) {
290 fBuffer[i] = 0;
291 for (int j = 0; j < fNumNus; j++) {
292 fBuffer[i] += EvecA(i, j) * fNuState[j];
293 }
294 fBuffer[i] *= fPhases[i];
295 }
296
297 // Propagate neutrino state
298 for (int i = 0; i < fNumNus; i++) {
299 fNuState[i] = 0;
300 for (int j = 0; j < fNumNus; j++) {
301 fNuState[i] += Evec(i, j) * fBuffer[j];
302 }
303 }
304}

326void PMNS_NUNM::UpdateHam()
327{
328 double rho = fPath.density;
329 double zoa = fPath.zoa;
330
331 double lv = 2 * kGeV2eV * fEnergy; // 2*E in eV
332
333 double kr2GNe =
334 kK2 * M_SQRT2 * kGf * rho * zoa; // Electron matter potential in eV
335 double kr2GNn = kK2 * M_SQRT2 * kGf * rho * (1. - zoa) / 2. *
336 fracVnc; // Neutron matter potential in eV
337
338 // Finish build Hamiltonian in matter with dimension of eV
339
340 for (int i = 0; i < fNumNus; i++) {
341 if (!fIsNuBar) { V(i, i) = -1. * kr2GNn; }
342 else {
343 V(i, i) = kr2GNn;
344 }
345 for (int j = 0; j < fNumNus; j++) {
346 if (!fIsNuBar)
347 Ham(i, j) = fHms[i][j] / lv;
348 else
349 Ham(i, j) = conj(fHms[i][j]) / lv;
350 }
351 }
352
353 if (!fIsNuBar) { V(0, 0) += kr2GNe; }
354 else {
355 V(0, 0) -= kr2GNe;
356 }
357
358 if (fscale == 0) {
359 Ham += Alpha.adjoint() * V * Alpha;
360 } // low scale scenario with mixing matrix part of larger unitary matrix
361 else if (fscale == 1) {
362 Ham += Alpha.inverse() * V * (Alpha.adjoint()).inverse();
363 } // high scale scenario with non unitary mixing matrix
364
365 for (int i = 0; i < fNumNus; i++) {
366 for (int j = 0; j < fNumNus; j++) { fHam[i][j] = Ham(i, j); }
367 }
