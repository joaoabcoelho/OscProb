165void PMNS_Decay::BuildHms()
166{
167
168 // Check if anything changed
169 if(fBuiltHms) return;
170
171 // Tag to recompute eigensystem
172 fGotES = false;
173
174 for(int j=0; j<fNumNus; j++){
175 // Set mass splitting
176 fHms[j][j] = fDm[j];
177 fHd[j][j] = fAlpha[j];
178 // Reset off-diagonal elements
179 for(int i=0; i<j; i++){
180 fHms[i][j] = 0;
181 fHd[i][j] = 0;
182 }
183 // Rotate j neutrinos
184 for(int i=0; i<j; i++){
185 RotateH(i,j,fHms);
186 RotateH(i,j,fHd);
187 }
188 }
189
190 // Taking care of antineutrinos delta-> -delta and filling the upper triangle
191 // because final hamiltonian will be non-hermitian.
192 for(int i=0; i<fNumNus; i++){
193 for(int j=i+1; j<fNumNus; j++){
194 if(fIsNuBar){
195 fHms[i][j] = conj(fHms[i][j]);
196 fHd[i][j] = conj(fHd[i][j]);
197 }
198 fHms[j][i] = conj(fHms[i][j]);
199 fHd[j][i] = conj(fHd[i][j]);
200 }
201 }
202
203 const complexD numi(0.0,1.0);
205 for(int j=0; j<fNumNus; j++){
206 for(int i=0; i<fNumNus; i++){
207 fHms[i][j] = fHms[i][j] - numi*fHd[i][j];
208 }
209 }
210
211 // Clear eigensystem cache
212 //ClearCache();
213
214 // Tag as built
215 fBuiltHms = true;
216
217}

228void PMNS_Decay::UpdateHam()
229{
230
231 double lv = 2 * kGeV2eV*fEnergy; // 2E in eV
232
233 double kr2GNe = kK2*M_SQRT2*kGf;
234 kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
235
236 // Finish building Hamiltonian in matter with dimension of eV
237 for(int i=0; i<fNumNus; i++){
238 for(int j=0; j<fNumNus; j++){
239 fHam(i,j) = fHms[i][j]/lv;
240 }
241 }
242
243 if(!fIsNuBar) fHam(0,0) += kr2GNe;
244 else fHam(0,0) -= kr2GNe;
245
246}

287void PMNS_Decay::PropagatePath(NuPath p)
288{
289
290 // Set the neutrino path
291 SetCurPath(p);
292
293 // Build Hamiltonian
294 BuildHms();
295 UpdateHam();
296
297 double LengthIneV = kKm2eV * p.length;
298
299 // Compute evolution operator exp(-I*H*L)
300 Eigen::Matrix3cd H = fHam;
301 H *= complexD(0, -LengthIneV);
302 H = H.exp();
303
304 // Propagate using evolution operator
305 for(int i=0;i<fNumNus;i++){
306 fBuffer[i] = 0;
307 for(int j=0;j<fNumNus;j++){
308 fBuffer[i] += H(i,j) * fNuState[j];
309 }
310 }
311
312 // Update neutrino state
313 for(int i=0;i<fNumNus;i++){
314 fNuState[i] = fBuffer[i];
315 }
316
317}
