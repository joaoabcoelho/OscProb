140void PMNS_Decay::BuildHam()
141{
142
143 // Check if anything changed
144 if(fBuiltHms) return;
145
146 // Tag to recompute eigensystem
147 fGotES = false;
148
150 for(int i=0; i<fNumNus; i++){
151 for(int j=0; j<fNumNus; j++){
152 fHms[i][j]= 0;
153 }
154 }
155 for(int j=0; j<fNumNus; j++){
156 // Set mass splitting
157 fHms[j][j] = fDm[j];
158 //Rotate j neutrinos
159 for(int i=0; i<j; i++){
160 RotateH(i,j,fHms);
161 }
162 }
163 //Taking care of antineutrinos delta-> -delta and filling the upper triangle
164 // because final hamiltonian will be non-hermitian.
165 for(int i=0; i<fNumNus; i++){
166 for(int j=i+1; j<fNumNus; j++){
167 if(fIsNuBar){
168 fHms[i][j]=conj(fHms[i][j]);
169 }
170 fHms[j][i]= conj(fHms[i][j]);
171 }
172 }
173
174
177 for(int i=0; i<fNumNus; i++){
178 for(int j=0; j<fNumNus; j++){
179 fHd[i][j]= 0;
180 }
181 }
182
183 for(int j=0; j<fNumNus; j++){
184
185 // Set alpha
186 fHd[j][j] = falpha[j];
187
188
189 // Rotate j neutrinos
190 for(int i=0; i<j; i++){
191 RotateH(i,j,fHd);
192 }
193 }
194 //Taking care of antineutrinos delta-> -delta and filling the upper triangle
195 // because final hamiltonian will be non-hermitian.
196 for(int i=0; i<fNumNus; i++){
197 for(int j=i+1; j<fNumNus; j++){
198 if(fIsNuBar){
199 fHd[i][j]=conj(fHd[i][j]);
200 }
201 fHd[j][i]= conj(fHd[i][j]);
202 }
203 }
204
205
206 const complex<double> numi(0.0,1.0);
208 for(int j=0; j<fNumNus; j++){
209 for(int i=0; i<fNumNus; i++){
210 fHt[i][j]=fHms[i][j]-numi*fHd[i][j];
211 }
212 }
213
214 // Clear eigensystem cache
215 //ClearCache();
216
217 // Tag as built
218 fBuiltHms = true;
219
220}

231void PMNS_Decay::UpdateHam()
232{
233
234 double lv = 2 * kGeV2eV*fEnergy; // 2E in eV
235
236 double kr2GNe = kK2*M_SQRT2*kGf;
237 kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
238
239 // Finish building Hamiltonian in matter with dimension of eV
240 for(int i=0;i<fNumNus;i++){
241 for(int j=0; j<fNumNus; j++){
242 fHam[i][j] = fHt[i][j]/lv;
243 }
244 }
245
246 if(!fIsNuBar) fHam[0][0] += kr2GNe;
247 else fHam[0][0] -= kr2GNe;
248
249}

257void PMNS_Decay::SolveHam()
258{
259
260 // Build Hamiltonian
261 BuildHam();
262
263 // Check if anything changed
264 if(fGotES) return;
265

273
274
275 // Fill fEval and fEvec vectors from Eigen arrays
276 for(int i=0;i<fNumNus;i++){
277 fEval[i] = fEvalEigen[i].real();
278 fEvalI[i] = fEvalEigen[i].imag();
279 for(int j=0;j<fNumNus;j++){
280 fEvec[i][j] = fEvecEigen[i][j];
281 fEvecinv[i][j] = fEvecEigeninv[i][j];
282 }

298void PMNS_Decay::PropagatePath(NuPath p)
299{
300
301 // Set the neutrino path
302 SetCurPath(p);
303
304 // Solve for eigensystem
305 SolveHam();
306
307 double LengthIneV = kKm2eV * p.length;
308 for(int i=0; i<fNumNus; i++){
309 complexD arg(fEvalI[i], -fEval[i]);
310 fPhases[i] = exp(arg * LengthIneV);
311 }
312 for(int i=0;i<fNumNus;i++){
313 fBuffer[i] = 0;
314 for(int j=0;j<fNumNus;j++){
315 fBuffer[i] += fEvecinv[i][j] * fNuState[j];
316 }
317 fBuffer[i] *= fPhases[i];
318 }
319
320 // Propagate neutrino state
321 for(int i=0;i<fNumNus;i++){
322 fNuState[i] = 0;
323 for(int j=0;j<fNumNus;j++){
324 fNuState[i] += fEvec[i][j] * fBuffer[j];
325 }
326 }
