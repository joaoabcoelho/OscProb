78 void PMNS_Sterile::SolveHam()
79 {
80 
81  // Build Hamiltonian
82  BuildHms();
83 
84  // Check if anything changed
85  if(fGotES) return;
86 
87  double rho = fPath.density;
88  double zoa = fPath.zoa;
89 
90  double lv = 2 * kGeV2eV*fEnergy; // 2E in eV
91  double kr2GNe = kK2*M_SQRT2*kGf * rho*zoa; // Electron matter potential in eV
92  double kr2GNn = kK2*M_SQRT2*kGf * rho*(1-zoa)/2; // Neutron matter potential in eV
93 
94  // Finish building Hamiltonian in matter with dimension of eV
95  for(size_t i=0;i<size_t(fNumNus);i++){
96  complex buf = fHms[i][i]/lv;
97  *gsl_matrix_complex_ptr(H_GSL, i, i) = gsl_complex_rect(real(buf),0);
98  for(size_t j=i+1;j<size_t(fNumNus);j++){
99  buf = fHms[i][j]/lv;
100  if(!fIsNuBar) *gsl_matrix_complex_ptr(H_GSL, j, i) = gsl_complex_rect(real(buf),-imag(buf));
101  else *gsl_matrix_complex_ptr(H_GSL, j, i) = gsl_complex_rect(real(buf), imag(buf));
102  }
103  if(i>2){
104  // Subtract NC coherent forward scattering from sterile neutrinos. See arXiv:hep-ph/0606054v3, eq. 3.15, for example.
105  if(!fIsNuBar) *gsl_matrix_complex_ptr(H_GSL, i, i) = gsl_complex_add_real(gsl_matrix_complex_get(H_GSL,i,i) , kr2GNn);
106  else *gsl_matrix_complex_ptr(H_GSL, i, i) = gsl_complex_add_real(gsl_matrix_complex_get(H_GSL,i,i) , -kr2GNn);;
107  }
108  }
109  // Add nue CC coherent forward scattering from sterile neutrinos.
110  if(!fIsNuBar) *gsl_matrix_complex_ptr(H_GSL, 0, 0) = gsl_complex_add_real(gsl_matrix_complex_get(H_GSL,0,0) , kr2GNe);
111  else *gsl_matrix_complex_ptr(H_GSL, 0, 0) = gsl_complex_add_real(gsl_matrix_complex_get(H_GSL,0,0) , -kr2GNe);
112 
113  // Solve Hamiltonian for eigensystem
114  gsl_eigen_hermv(H_GSL, fEvalGSL, fEvecGSL, W_GSL);
115 
116  // Fill fEval and fEvec vectors from GSL objects
117  for(int i=0;i<fNumNus;i++){
118  fEval[i] = gsl_vector_get(fEvalGSL,i);
119  for(int j=0;j<fNumNus;j++){
120  gsl_complex buf = gsl_matrix_complex_get(fEvecGSL,i,j);
121  fEvec[i][j] = complex( GSL_REAL(buf), GSL_IMAG(buf) );
122  }
123  }
124 
125  fGotES = true;
126 
127 }
