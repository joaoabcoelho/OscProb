76 void PMNS_Fast::UpdateHam()
77 {
78 
79  double lv = 2 * kGeV2eV*fEnergy; // 2E in eV
80 
81  double kr2GNe = kK2*M_SQRT2*kGf;
82  kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
83 
84  // Finish building Hamiltonian in matter with dimension of eV
85  for(int i=0;i<fNumNus;i++){
86  fHam[i][i] = fHms[i][i]/lv;
87  for(int j=i+1;j<fNumNus;j++){
88  if(!fIsNuBar) fHam[i][j] = fHms[i][j]/lv;
89  else fHam[i][j] = conj(fHms[i][j])/lv;
90  }
91  }
92  if(!fIsNuBar) fHam[0][0] += kr2GNe;
93  else fHam[0][0] -= kr2GNe;
94 
95 }

152 void PMNS_Fast::SetVacuumEigensystem()
153 {
154 
155  double s12, s23, s13, c12, c23, c13;
156  complexD idelta(0.0, fDelta[0][2]);
157  if(fIsNuBar) idelta = conj(idelta);
158 
159  s12 = sin(fTheta[0][1]); s23 = sin(fTheta[1][2]); s13 = sin(fTheta[0][2]);
160  c12 = cos(fTheta[0][1]); c23 = cos(fTheta[1][2]); c13 = cos(fTheta[0][2]);
161 
162  fEvec[0][0] = c12*c13;
163  fEvec[0][1] = s12*c13;
164  fEvec[0][2] = s13*exp(-idelta);
165 
166  fEvec[1][0] = -s12*c23-c12*s23*s13*exp(idelta);
167  fEvec[1][1] = c12*c23-s12*s23*s13*exp(idelta);
168  fEvec[1][2] = s23*c13;
169 
170  fEvec[2][0] = s12*s23-c12*c23*s13*exp(idelta);
171  fEvec[2][1] = -c12*s23-s12*c23*s13*exp(idelta);
172  fEvec[2][2] = c23*c13;
173 
174  fEval[0] = 0;
175  fEval[1] = fDm[1] / (2 * kGeV2eV*fEnergy);
176  fEval[2] = fDm[2] / (2 * kGeV2eV*fEnergy);
177 
178 }
