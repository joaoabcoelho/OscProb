251  double rBelow = fPremLayers[fDetLayer].radius; // Radius below detector

291 int PremModel::FillPath(double cosT)
292 {
293 
294  // Clear current path sequence
295  fNuPath.clear();
296 
297  // Do nothing if cosine is unphysical
298  if(fabs(cosT) > 1) return 0;
299 
300  // Define the minimum path radius
301  double minR = fPremLayers[fDetLayer].radius * sqrt(1 - cosT*cosT);
302 
303  // Set the top layer index
304  int toplayer = fPremLayers.size() - 1;
305 
306  // Find the inner-most crossed layer
307  int minlayer = 0;
308  while(fPremLayers[minlayer].radius < minR) minlayer++;
309 
310  // Compute the number of path segments needed
311  int nsteps = toplayer - fDetLayer;
312  if(cosT < 0) nsteps += 2*(fDetLayer - minlayer) + 1;
313 
314  // Start at the top layer and go down
315  int layer = toplayer;
316  int dl = -1;
317 
318  // Loop over all path segments
319  for(int i=0; i<nsteps; i++){
320 
321  // Get square of the path length between this layer's
322  // outer radius and inner-most radius
323  double L1 = pow(fPremLayers[layer].radius,2) - minR*minR;
324 
325  // If L1 is negative, outer radius is not crossed.
326  // This only happens if detector is at the top layer and the
327  // neutrino is coming from above.
328  if(L1 < 0) return true;
329 
330  // Get square of the path length between this layer's
331  // inner radius and inner-most radius
332  double L2 = -minR*minR;
333  if(layer>0) L2 += pow(fPremLayers[layer-1].radius,2);
334 
335  // If L2 is negative, inner radius is not crossed,
336  // so set this as the minimum layer.
337  bool ismin = (L2<=0 && cosT < 0);
338 
339  // Store the path segment length
340  double dL;
341 
342  // If it's the minimum layer, connect two outer radius
343  // crossing points. If not, compute difference between
344  // inner and outer radius crossings.
345  if(ismin) dL = 2 * sqrt(L1);
346  else if(L2>=0) dL = sqrt(L1) - sqrt(L2);
347  else dL = sqrt(L1); // This should never actually happen,
348  // but protect in case L2 is slightly
349  // negative when it should be zero.
350  // e.g. arriving at the detector from above.
351 
352  // Add this path segment to the sequence
353  AddPath(dL, fPremLayers[layer]);
354 
355  // If we reached the inner-most layer,
356  // start moving up again.
357  if(ismin) dl = 1;
358 
359  // Move to next layer
360  layer += dl;
361 
362  }
363 
364  // Return the number of path segments
365  return fNuPath.size();
366 
367 }
