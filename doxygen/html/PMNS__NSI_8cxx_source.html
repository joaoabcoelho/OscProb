226 void PMNS_NSI::UpdateHam()
227 {
228 
229  double lv = 2 * kGeV2eV*fEnergy; // 2*E in eV
230 
231  double kr2GNe = kK2*M_SQRT2*kGf * fPath.density;
232  double kr2GNnsi = kr2GNe;
233 
234  kr2GNe *= fPath.zoa; // Std matter potential in eV
235  kr2GNnsi *= GetZoACoup(); // NSI matter potential in eV
236 
237  // Finish build Hamiltonian in matter with dimension of eV
238  for(int i=0;i<fNumNus;i++){
239  for(int j=i;j<fNumNus;j++){
240  if(!fIsNuBar) fHam[i][j] = fHms[i][j]/lv + kr2GNnsi*fEps[i][j];
241  else fHam[i][j] = conj(fHms[i][j]/lv - kr2GNnsi*fEps[i][j]);
242  }
243  }
244 
245  if(!fIsNuBar) fHam[0][0] += kr2GNe;
246  else fHam[0][0] -= kr2GNe;
247 
248 }
