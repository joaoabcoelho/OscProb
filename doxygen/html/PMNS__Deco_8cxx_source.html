80 void PMNS_Deco::SetGamma32(double val){
81 
82  double min32 = 0.25*fGamma[1];
83 
84  if(fGamma[0] >= 0) min32 *= 1 - pow(fGamma[0],2);
85  else min32 *= 1 + 3*pow(fGamma[0],2);
86 
87  if(val < min32){
88 
89  if(fGamma[0]>=0 || 4*val/fGamma[1] < 1){
90  fGamma[0] = sqrt(1 - 4*val/fGamma[1]);
91  fGamma[2] = 0.25*fGamma[1]*(1 + 3*pow(fGamma[0],2));
92  }
93  else{
94  fGamma[0] = -sqrt((4*val/fGamma[1] - 1)/3);
95  fGamma[2] = 0.25*fGamma[1]*(1 - pow(fGamma[0],2));
96  }
97 
98  cout << "WARNING: Impossible to have Gamma32 = " << val
99  << " with current Gamma21 and theta parameters." << endl
100  << " Changing the value of cos(theta) to " << fGamma[0]
101  << endl;
102 
103  return;
104 
105  }
106 
107  double arg = fGamma[1] * ( 4*val - fGamma[1]*(1 - pow(fGamma[0],2)) );
108 
109  if(arg < 0){
110  arg = 0;
111  fGamma[0] = sqrt(1 - 4*val/fGamma[1]);
112  cout << "WARNING: Imaginary Gamma31 found. Changing the value of cos(theta) to "
113  << fGamma[0] << endl;
114  }
115 
116  double gamma31 = val + fGamma[1]*pow(fGamma[0],2) + fGamma[0] * sqrt(arg);
117 
118  fGamma[2] = gamma31;
119 
120  if(fabs(val - GetGamma(3,2)) > 1e-6){
121  cout << "ERROR: Failed sanity check: GetGamma(3,2) = "
122  << GetGamma(3,2) << " != " << val << endl;
123  }
124 
125 }

195 double PMNS_Deco::GetGamma(int i, int j)
196 {
197

201  int temp = i;

206  cout << "Gamma_" << i << j << " not valid for " << fNumNus;

209  }
210 
211  if(j == 1){
212  return fGamma[i-1];
213  }
214  else {

216  double arg = fGamma[1] * ( 4*fGamma[2] - fGamma[1]*(1 - pow(fGamma[0],2)) );
217

304 void PMNS_Deco::PropagatePath(NuPath p)
305 {
306 
307  // Set the neutrino path
308  SetCurPath(p);
309 
310  // Solve for eigensystem
311  SolveHam();
312 
313  // The code below is not pretty or optimised at all. Lots of matrix
314  // multiplications are going on which could be otpimised due to
315  // symmetry. The index sorting is also terrible. Needs improving.
316 
317  // Store rotation matrices
318  matrix U = fEvec;
319  matrix Ut = CTransp(fEvec);
320 
321  // Rotate to effective mass basis
322  fRho = Dot(Ut, Dot(fRho, U));
323 
324  // Compute evolution matrix
325  matrix evolve(3, row(3,1));
326 
327  // Some ugly way of matching gamma and dmsqr indices
328  int maxdm = 0;
329  if(fDm[1] > fDm[maxdm]) maxdm = 1;
330  if(fDm[2] > fDm[maxdm]) maxdm = 2;
331 
332  int mindm = 0;
333  if(fDm[1] < fDm[mindm]) mindm = 1;
334  if(fDm[2] < fDm[mindm]) mindm = 2;
335 
336  int middm = 0;
337  if(middm == maxdm || middm == mindm) middm = 1;
338  if(middm == maxdm || middm == mindm) middm = 2;
339 
340  int maxev = 0;
341  if(fEval[1] > fEval[maxev]) maxev = 1;
342  if(fEval[2] > fEval[maxev]) maxev = 2;
343 
344  int minev = 0;
345  if(fEval[1] < fEval[minev]) minev = 1;
346  if(fEval[2] < fEval[minev]) minev = 2;
347 
348  int midev = 0;
349  if(midev == maxev || midev == minev) midev = 1;
350  if(midev == maxev || midev == minev) midev = 2;
351 
352  int idx[3];
353  idx[minev] = mindm;
354  idx[midev] = middm;
355  idx[maxev] = maxdm;
356 
357  double energyCorr = pow(fEnergy, fPower);
358 
359  for(int j=0; j<3; j++){
360  for(int i=0; i<j; i++){
361  double gamma_ij = GetGamma(max(idx[i],idx[j])+1, min(idx[i],idx[j])+1) * energyCorr;
362  complexD arg = complexD(gamma_ij * kGeV2eV, fEval[i] - fEval[j]) * kKm2eV * p.length;
363  evolve[i][j] = exp(-arg);
364  evolve[j][i] = exp(-conj(arg));
365  }}
366 
367  // Evolve density matrix
368  fRho = Mult(fRho, evolve);
369 
370  // Rotate back to flavour basis
371  fRho = Dot(U, Dot(fRho, Ut));
372 
373 }
