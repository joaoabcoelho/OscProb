111void PMNS_LIV::SetaT(int flvi, int flvj, double val, double phase){
112
113 if(flvi > flvj){
114 cout << "First argument should be smaller or equal to second argument" << endl;
115 cout << "Setting reverse order (aT_" << flvj << flvi << "). " << endl;
116 int temp = flvi;
117 flvi = flvj;
118 flvj = temp;
119 }
120 if(flvi<0 || flvi>2 || flvj < flvi || flvj > 2){
121 cout << "aT_" << flvi << flvj << " not valid for " << fNumNus;
122 cout << " neutrinos. Doing nothing." << endl;
123 return;
124 }
125
126 complexD h = val;
127
128 if(flvi != flvj) h *= complexD(cos(phase), sin(phase));
129
130 bool isSame = (faT[flvi][flvj] == h);
131
132 if(!isSame) ClearCache();
133
134 fGotES *= isSame;
135
136 faT[flvi][flvj] = h;
137
138}

361void PMNS_LIV::UpdateHam()
362{
363
364 double lv = 2 * kGeV2eV*fEnergy; // 2*E in eV
365 //double fEnergy_ev = kGeV2eV*fEnergy; // E in eV
366
367 double kr2GNe = kK2*M_SQRT2*kGf*fPath.density;
368 kr2GNe *= fPath.zoa; // Matter potential in eV
369
370 // Finish build Hamiltonian in matter with dimension of eV
371 for(int i=0;i<fNumNus;i++){
372 for(int j=i;j<fNumNus;j++){
373 if(!fIsNuBar) fHam[i][j] = fHms[i][j]/lv + kGeV2eV*(faT[i][j] - (4.*fEnergy / 3.)*fcT[i][j]);
374 else fHam[i][j] = conj(fHms[i][j]/lv + kGeV2eV*(- faT[i][j] - (4.*fEnergy / 3.)*fcT[i][j])); // +/- faTs for nu/nubar (CPTodd)
375 }
376 }
377
378 if(!fIsNuBar) fHam[0][0] += kr2GNe;
379 else fHam[0][0] -= kr2GNe;
380
381}
