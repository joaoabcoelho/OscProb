67void PMNS_LIV::SetaT(int flvi, int flvj, int dim, double val, double phase){
68
69 if(flvi > flvj){
70 cerr << "WARNING: First argument should be smaller or equal to second argument" << endl
71 << "Setting reverse order (aT_" << flvj << flvi << "). " << endl;
72 int temp = flvi;
73 flvi = flvj;
74 flvj = temp;
75 }
76 if(flvi<0 || flvi>2 || flvj < flvi || flvj > 2){
77 cerr << "WARNING: aT_" << flvi << flvj << " not valid for " << fNumNus
78 << " neutrinos. Doing nothing." << endl;
79 return;
80 }
81 if(dim != 3 && dim != 5 && dim != 7){
82 cerr << "WARNING: Invalid aT coefficient dimension dim=" << dim << " not in [3,5,7].\n";
83 return;
84 }
85
86 int pos = (dim-3)/2;
87
88 complexD h = val;
89
90 if(flvi != flvj) h *= complexD(cos(phase), sin(phase));
91
92 bool isSame = (faT[flvi][flvj][pos] == h);
93
94 if(!isSame) ClearCache();
95
96 fGotES *= isSame;
97
98 faT[flvi][flvj][pos] = h;
99
100}

239void PMNS_LIV::UpdateHam()
240{
241
242 double lv = 2 * kGeV2eV*fEnergy; // 2*E in eV
243
244 // Set the vacuum Hamiltonian
245 for(int i=0;i<fNumNus;i++){
246 for(int j=i;j<fNumNus;j++){
247 fHam[i][j] = fHms[i][j]/lv;
248 }}
249
250 // Add matter potential
251 double kr2GNe = kK2*M_SQRT2*kGf * fPath.density * fPath.zoa;
252 if(!fIsNuBar) fHam[0][0] += kr2GNe;
253 else fHam[0][0] -= kr2GNe;
254
255
256 // Add LIV terms for all dimensions
257 double energy_pow = kGeV2eV;
258
259 for(int dim = 3; dim < 9; dim++){
260
261 // Increase energy power
262 if(dim>3) energy_pow *= fEnergy;
263
264 // Set parameter index
265 int coeff_idx = (dim-3)/2;
266
267 for(int i=0;i<fNumNus;i++){
268 for(int j=i;j<fNumNus;j++){
269
270 complexD liv_term = energy_pow;
271
272 if(dim%2==1){ // aT is CPT-odd
273 if(fIsNuBar) liv_term *= -faT[i][j][coeff_idx];
274 else liv_term *= faT[i][j][coeff_idx];
275 }
276 else { // cT is CPT-even
277 // Add 4/3 coefficient for backward compatibility
278 if(dim==4) liv_term *= -4/3. * fcT[i][j][coeff_idx];
279 else liv_term *= -fcT[i][j][coeff_idx];
280 }
281
282 fHam[i][j] += liv_term;
283
284 }}
285
286 }
287
288 // Conjugate Hamiltonian for antineutrinos
289 if(fIsNuBar){
290 for(int i=0;i<fNumNus;i++){
291 for(int j=i+1;j<fNumNus;j++){
292 fHam[i][j] = conj(fHam[i][j]);
293 }}
294 }
295
296}
