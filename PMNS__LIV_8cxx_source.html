67void PMNS_LIV::SetaT(int flvi, int flvj, int dim, double val, double phase){
68
69 if(flvi > flvj){
70 cerr << "WARNING: First argument should be smaller or equal to second argument" << endl
71 << "Setting reverse order (aT_" << flvj << flvi << "). " << endl;
72 int temp = flvi;
73 flvi = flvj;
74 flvj = temp;
75 }
76 if(flvi<0 || flvi>2 || flvj < flvi || flvj > 2){
77 cerr << "WARNING: aT_" << flvi << flvj << " not valid for " << fNumNus
78 << " neutrinos. Doing nothing." << endl;
79 return;
80 }
81 if(dim != 3 && dim != 5 && dim != 7){
82 cerr << "WARNING: Invalid aT coefficient dimension dim=" << dim << " not in [3,5,7].\n";
83 return;
84 }
85
86 int pos = (dim-3)/2;
87
88 complexD h = val;
89
90 if(flvi != flvj) h *= complexD(cos(phase), sin(phase));
91
92 bool isSame = (faT[flvi][flvj][pos] == h);
93
94 if(!isSame) ClearCache();
95
96 fGotES *= isSame;
97
98 faT[flvi][flvj][pos] = h;
99
100}

239void PMNS_LIV::UpdateHam()
240{
241
242 double lv = 2 * kGeV2eV*fEnergy; // 2*E in eV
243
244 double kr2GNe = kK2*M_SQRT2*kGf*fPath.density;
245 kr2GNe *= fPath.zoa; // Matter potential in eV
246
247 // Finish build Hamiltonian in matter with dimension of eV
248 for(int i=0;i<fNumNus;i++){
249 for(int j=i;j<fNumNus;j++){
250 complexD liv_term = 0;
251 for(int dim = 3; dim < 9; dim++){
252 complexD dim_term = std::pow(fEnergy, dim-3);
253 if(dim%2==1) dim_term *= (fIsNuBar ? -1. : 1.) * faT[i][j][(dim-3)/2];
254 else dim_term *= (dim==4 ? -4/3. : -1.) * fcT[i][j][(dim-4)/2];
255 liv_term += dim_term;
256 }
257
258 fHam[i][j] = fHms[i][j]/lv + kGeV2eV * liv_term;
259
260 if(fIsNuBar) fHam[i][j] = conj(fHam[i][j]);
261 }
262 }
263
264 if(!fIsNuBar) fHam[0][0] += kr2GNe;
265 else fHam[0][0] -= kr2GNe;
266
267}
