67void PMNS_LIV::SetaT(int flvi, int flvj, int dim, double val, double phase)
68{
69 if (flvi > flvj) {
70 cerr << "WARNING: First argument should be smaller or equal to second "
71 "argument"
72 << endl
73 << "Setting reverse order (aT_" << flvj << flvi << "). " << endl;
74 int temp = flvi;
75 flvi = flvj;
76 flvj = temp;
77 }
78 if (flvi < 0 || flvi > 2 || flvj < flvi || flvj > 2) {
79 cerr << "WARNING: aT_" << flvi << flvj << " not valid for " << fNumNus
80 << " neutrinos. Doing nothing." << endl;
81 return;
82 }
83 if (dim != 3 && dim != 5 && dim != 7) {
84 cerr << "WARNING: Invalid aT coefficient dimension dim=" << dim
85 << " not in [3,5,7].\n";
86 return;
87 }
88
89 int pos = (dim - 3) / 2;
90
91 complexD h = val;
92
93 if (flvi != flvj) h *= complexD(cos(phase), sin(phase));
94
95 bool isSame = (faT[flvi][flvj][pos] == h);
96
97 if (!isSame) ClearCache();
98
99 fGotES *= isSame;
100
101 faT[flvi][flvj][pos] = h;
102}

254void PMNS_LIV::UpdateHam()
255{
256 double lv = 2 * kGeV2eV * fEnergy; // 2*E in eV
257
258 // Set the vacuum Hamiltonian
259 for (int i = 0; i < fNumNus; i++) {
260 for (int j = i; j < fNumNus; j++) { fHam[i][j] = fHms[i][j] / lv; }
261 }
262
263 // Add matter potential
264 double kr2GNe = kK2 * M_SQRT2 * kGf * fPath.density * fPath.zoa;
265 if (!fIsNuBar)
266 fHam[0][0] += kr2GNe;
267 else
268 fHam[0][0] -= kr2GNe;
269
270 // Add LIV terms for all dimensions
271 double energy_pow = kGeV2eV;
272
273 for (int dim = 3; dim < 9; dim++) {
274 // Increase energy power
275 if (dim > 3) energy_pow *= fEnergy;
276
277 // Set parameter index
278 int coeff_idx = (dim - 3) / 2;
279
280 for (int i = 0; i < fNumNus; i++) {
281 for (int j = i; j < fNumNus; j++) {
282 complexD liv_term = energy_pow;
283
284 if (dim % 2 == 1) { // aT is CPT-odd
285 if (fIsNuBar)
286 liv_term *= -faT[i][j][coeff_idx];
287 else
288 liv_term *= faT[i][j][coeff_idx];
289 }
290 else { // cT is CPT-even
291 // Add 4/3 coefficient for backward compatibility
292 if (dim == 4)
293 liv_term *= -4 / 3. * fcT[i][j][coeff_idx];
294 else
295 liv_term *= -fcT[i][j][coeff_idx];
296 }
297
298 fHam[i][j] += liv_term;
299 }
300 }
301 }
302
303 // Conjugate Hamiltonian for antineutrinos
304 if (fIsNuBar) {
305 for (int i = 0; i < fNumNus; i++) {
306 for (int j = i + 1; j < fNumNus; j++) { fHam[i][j] = conj(fHam[i][j]); }
307 }
308 }
309}
